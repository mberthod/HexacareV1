1/**
2 * @file panic_handler.c
3 * @ingroup group_storage
4 * @brief Capture des paniques système et écriture dans LittleFS.
5 *
6 * Architecture :
7 * Le système ESP-IDF ne permet pas un accès fiable au système de fichiers
8 * pendant une panique matérielle (le heap peut être corrompu). La stratégie
9 * adoptée est donc à deux niveaux :
10 *
11 * Niveau 1 — shutdown handler (redémarrages propres via esp_restart()) :
12 * Appelé avant le reset. Écrit un message de redémarrage dans panic.log.
13 *
14 * Niveau 2 — RTC memory (paniques matérielles, watchdog) :
15 * Lors d'une panique, un flag et un message courts sont stockés en
16 * RTC_DATA_ATTR (survit au reset sans coupure secteur).
17 * Au prochain boot, panic_handler_check_on_boot() lit ce flag et
18 * l'écrit dans panic.log avant d'effacer le flag.
19 *
20 * Note : CONFIG_ESP_SYSTEM_PANIC_PRINT_REBOOT=y dans sdkconfig.defaults
21 * garantit que le backtrace complet est imprimé sur la console UART
22 * avant le redémarrage — complément indispensable pour le débogage.
23 */
24

36/* ================================================================
37 * Variables RTC — survivent au reset sans coupure secteur
38 * ================================================================ */
39static RTC_DATA_ATTR uint32_t s_panic_magic = 0;
40static RTC_DATA_ATTR char s_panic_msg[192];
41static RTC_DATA_ATTR int64_t s_panic_time_us;
42
43#define PANIC_MAGIC_VALUE 0xDEADCAFEU
44
45/* ================================================================
46 * shutdown_handler (interne)
47 * @brief Appelée par esp_restart() — redémarrage propre.
48 *
49 * Positionne le flag RTC avec un message de redémarrage volontaire.
50 * La trame sera écrite dans panic.log au prochain boot.
51 * ================================================================ */
52static void shutdown_handler(void)
53{
54 s_panic_magic = PANIC_MAGIC_VALUE;
55 s_panic_time_us = esp_timer_get_time();
56 strncpy(s_panic_msg, "Redémarrage propre via esp_restart()", sizeof(s_panic_msg) - 1);
57 s_panic_msg[sizeof(s_panic_msg) - 1] = '\0';
58}
59
60/* ================================================================
61 * panic_handler_install
62 * @brief Installe le shutdown handler et initialise les variables RTC.
63 *
64 * @return ESP_OK si l'installation réussit.
65 * ================================================================ */
66esp_err_t panic_handler_install(void)
67{
68 ESP_ERROR_CHECK(esp_register_shutdown_handler(shutdown_handler));
69 ESP_LOGI(TAG, "Shutdown handler installé");
70 return ESP_OK;
71}

73/* ================================================================
74 * panic_handler_check_on_boot
75 * @brief Vérifie le flag RTC et écrit l'info dans panic.log si présent.
76 *
77 * Doit être appelée APRÈS littlefs_manager_init() pour que le
78 * système de fichiers soit disponible.
79 * ================================================================ */
80void panic_handler_check_on_boot(void)
81{
82 if (s_panic_magic != PANIC_MAGIC_VALUE) {
83 /* Aucun crash précédent détecté */
84 return;
85 }
86
87 ESP_LOGW(TAG, "Crash détecté lors du précédent boot — écriture dans panic.log");
88
89 /* Formatage du message avec horodatage */
90 char entry[256];
91 double secs = (double)s_panic_time_us / 1e6;
92 snprintf(entry, sizeof(entry),
93 "[%10.3f] PANIC: %s\n",
94 secs, s_panic_msg);
95
96 /* Écriture bloquante/synchrone dans panic.log */
97 FILE *f = fopen(PANIC_FILE_PATH, "a");
98 if (f) {
99 fwrite(entry, 1, strlen(entry), f);
100 fflush(f);
101 fclose(f);
102 ESP_LOGI(TAG, "Entrée de panique écrite dans %s", PANIC_FILE_PATH);
103 } else {
104 ESP_LOGE(TAG, "Impossible d'ouvrir %s", PANIC_FILE_PATH);
105 }
106
107 /* Effacement du flag pour ne pas réécrire au prochain boot */
108 s_panic_magic = 0;
109 memset(s_panic_msg, 0, sizeof(s_panic_msg));
110}
