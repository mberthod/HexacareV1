1/**
2 * @file log_writer.c
3 * @ingroup group_storage
4 * @brief Log circulaire sur LittleFS — rotation automatique à 900 Ko.
5 *
6 * Algorithme de rotation :
7 * 1. Ouvrir le fichier en mode "a" (append).
8 * 2. Écrire le message horodaté.
9 * 3. Si ftell() dépasse LOG_MAX_SIZE_BYTES :
10 * a. Lire tout le fichier en mémoire (allocation heap).
11 * b. Calculer le nombre d'octets à ignorer (LOG_TRIM_PERCENT %).
12 * c. Avancer jusqu'à la fin de la ligne correspondante.
13 * d. Réécrire le reste du fichier depuis le début.
14 *
15 * Thread-safety : mutex FreeRTOS créé à la première écriture.
16 */

30/* ================================================================
31 * log_trim_file (interne)
32 * @brief Supprime les LOG_TRIM_PERCENT % d'entrées les plus anciennes.
33 *
34 * Appelée uniquement depuis log_write() quand le fichier dépasse
35 * LOG_MAX_SIZE_BYTES. Alloue un buffer temporaire sur le tas.
36 *
37 * @param file_size Taille actuelle du fichier en octets.
38 * ================================================================ */
39static void log_trim_file(long file_size)
40{
41 FILE *f = fopen(LOG_FILE_PATH, "r");
42 if (!f) {
43 ESP_LOGW(TAG, "Impossible d'ouvrir %s pour rotation", LOG_FILE_PATH);
44 return;
45 }
46
47 /* Nombre d'octets à ignorer (10 % les plus anciens) */
48 long skip_target = file_size / LOG_TRIM_PERCENT;
49 long skipped = 0;
50 char line[256];
51
52 /* Avancer jusqu'à la fin de la ligne qui dépasse skip_target */
53 while (skipped < skip_target) {
54 if (fgets(line, sizeof(line), f) == NULL) {
55 break;
56 }
57 skipped += (long)strlen(line);
58 }
59
60 long rest_start = ftell(f);
61 long rest_size = file_size - rest_start;
62
63 if (rest_size <= 0) {
64 fclose(f);
65 remove(LOG_FILE_PATH);
66 ESP_LOGI(TAG, "Rotation : fichier vide après suppression, recréé");
67 return;
68 }
69
70 /* Lire le contenu restant */
71 char *buf = malloc((size_t)rest_size + 1);
72 if (!buf) {
73 fclose(f);
74 ESP_LOGE(TAG, "Rotation : allocation mémoire échouée (%ld octets)", rest_size);
75 return;
76 }
77
78 size_t read_bytes = fread(buf, 1, (size_t)rest_size, f);
79 fclose(f);
80
81 /* Réécrire le fichier sans les entrées supprimées */
82 f = fopen(LOG_FILE_PATH, "w");
83 if (f) {
84 fwrite(buf, 1, read_bytes, f);
85 fclose(f);
86 }
87
88 free(buf);
89 ESP_LOGI(TAG, "Rotation du log : %ld octets anciens supprimés, %ld conservés",
90 rest_start, (long)read_bytes);
91}

93/* ================================================================
94 * log_write
95 * @brief Écrit un message horodaté dans le log circulaire LittleFS.
96 *
97 * Format d'une entrée : "[XXXXXX.XXX] message\n"
98 * où XXXXXX.XXX = secondes depuis le boot (esp_timer_get_time / 1e6).
99 *
100 * @param msg Message à journaliser (chaîne C terminée par '\0').
101 * @return ESP_OK, ESP_ERR_TIMEOUT ou ESP_FAIL.
102 * ================================================================ */
103esp_err_t log_write(const char *msg)
104{
105 if (!msg) {
106 return ESP_ERR_INVALID_ARG;
107 }
108
109 /* Création du mutex lors du premier appel (init paresseuse) */
110 if (!s_mutex) {
111 s_mutex = xSemaphoreCreateMutex();
112 if (!s_mutex) {
113 return ESP_ERR_NO_MEM;
114 }
115 }
116
117 if (xSemaphoreTake(s_mutex, pdMS_TO_TICKS(200)) != pdTRUE) {
118 return ESP_ERR_TIMEOUT;
119 }
120
121 /* Horodatage en secondes depuis le boot */
122 int64_t us = esp_timer_get_time();
123 double secs = (double)us / 1e6;
124
125 FILE *f = fopen(LOG_FILE_PATH, "a");
126 if (!f) {
127 xSemaphoreGive(s_mutex);
128 return ESP_FAIL;
129 }
130
131 fprintf(f, "[%10.3f] %s\n", secs, msg);
132 long current_pos = ftell(f);
133 fclose(f);
134
135 /* Déclenchement de la rotation si le seuil est dépassé */
136 if (current_pos >= (long)LOG_MAX_SIZE_BYTES) {
137 log_trim_file(current_pos);
138 }
139
140 xSemaphoreGive(s_mutex);
141 return ESP_OK;
142}
