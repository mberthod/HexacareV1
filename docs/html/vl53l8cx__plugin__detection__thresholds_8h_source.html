55/**
56 * @brief The following macro are used to define the 'type' of a checker.
57 * They indicate the window of measurements, defined by low and a high
58 * thresholds.
59 */
60
61#define VL53L8CX_IN_WINDOW ((uint8_t)0U)
62#define VL53L8CX_OUT_OF_WINDOW ((uint8_t)1U)
63#define VL53L8CX_LESS_THAN_EQUAL_MIN_CHECKER ((uint8_t)2U)
64#define VL53L8CX_GREATER_THAN_MAX_CHECKER ((uint8_t)3U)
65#define VL53L8CX_EQUAL_MIN_CHECKER ((uint8_t)4U)
66#define VL53L8CX_NOT_EQUAL_MIN_CHECKER ((uint8_t)5U)
67
68/**
69 * @brief The following macro are used to define multiple checkers in the same
70 * zone, using operators. Please note that the first checker MUST always be a OR
71 * operation.
72 */
73
74#define VL53L8CX_OPERATION_NONE ((uint8_t)0U)
75#define VL53L8CX_OPERATION_OR ((uint8_t)0U)
76#define VL53L8CX_OPERATION_AND ((uint8_t)2U)
77
78/**
79 * @brief Structure VL53L8CX_DetectionThresholds contains a single threshold.
80 * This structure is never used alone, it must be used as an array of 64
81 * thresholds (defined by macro VL53L8CX_NB_THRESHOLDS).
82 */
83
84typedef struct {
85
86 /* Low threshold */
87 int32_t param_low_thresh;
88 /* High threshold */
89 int32_t param_high_thresh;
90 /* Measurement to catch (VL53L8CX_MEDIAN_RANGE_MM,...)*/
91 uint8_t measurement;
92 /* Windows type (VL53L8CX_IN_WINDOW, VL53L8CX_OUT_WINDOW, ...) */
93 uint8_t type;
94 /* Zone id. Please read VL53L5 user manual to find the zone id.Set macro
95 * VL53L8CX_LAST_THRESHOLD to indicates the end of checkers */
96 uint8_t zone_num;
97 /* Mathematics operation (AND/OR). The first threshold is always OR.*/
98 uint8_t mathematic_operation;
99}VL53L8CX_DetectionThresholds;

146/**
147 * @brief This function is used to enable or disable the auto-stop feature.
148 * When ToF runs in autonomous mode with detection threshold, the sensor
149 * only emits an interrupt (INT pin) when a threshold is reached. Interrupt
150 * is raised when the measurement is completed. It is possible to abort the ranging
151 * without waiting for end of measurement completed by enabling the auto-stop. The
152 * sensor emits an interrupt and quickly aborts the measurements in progress. Please
153 * note that vl53l8cx_stop_ranging() function needs to be used after interrupt raised
154 * for a clean stop.
155 * This function is used to get the auto_stop flag.
156 * @param (VL53L8CX_Configuration) *p_dev : VL53L8CX configuration structure.
157 * @param (uint8_t) *p_auto_stop : Pointer of auto-stop feature, 0 disabled
158 * (default) or 1 enabled.
159 * @return (uint8_t) status : 0 if programming is OK
160 */
161
162uint8_t vl53l8cx_get_detection_thresholds_auto_stop(
163 VL53L8CX_Configuration *p_dev,
164 uint8_t *p_auto_stop);
