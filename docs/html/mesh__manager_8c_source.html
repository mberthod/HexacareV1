3 * @ingroup group_mesh
4 * @brief Gestionnaire réseau maillé ESP-NOW — topologie arborescente + AES-128.
5 *
6 * Chiffrement applicatif (contrainte +1000 nœuds) :
7 * Le LMK ESP-NOW est limité à ~17 pairs simultanés.

116/* ================================================================
117 * crc16_ccitt (interne)
118 * @brief Calcule le CRC16-CCITT d'un buffer.
119 *
120 * @param data Pointeur vers les données.
121 * @param length Longueur en octets.
122 * @return CRC16 calculé.
123 * ================================================================ */
124static uint16_t crc16_ccitt(const uint8_t *data, size_t length)
125{
126 uint16_t crc = 0xFFFF;
127 for (size_t i = 0; i < length; i++) {
128 crc ^= (uint16_t)data[i] << 8;
129 for (int j = 0; j < 8; j++) {
130 if (crc & 0x8000) {
131 crc = (uint16_t)((crc << 1) ^ 0x1021);
132 } else {
133 crc <<= 1;
134 }
135 }
136 }
137 return crc;
138}

140/* ================================================================
141 * aes_encrypt_payload (interne)
142 * @brief Chiffre un buffer en AES-128-CBC via mbedTLS.
143 *
144 * Le padding PKCS#7 est appliqué pour aligner sur 16 octets.
145 * L'IV est généré aléatoirement (esp_fill_random).
146 *
147 * @param plain Données en clair.
148 * @param plain_len Longueur des données en clair.
149 * @param cipher Buffer chiffré (doit être >= padded_len).
150 * @param iv Buffer IV (16 octets, rempli par cette fonction).
151 * @return Longueur du payload chiffré (multiple de 16), ou 0 si erreur.
152 * ================================================================ */
153static size_t aes_encrypt_payload(const uint8_t *plain, size_t plain_len,
154 uint8_t *cipher, uint8_t *iv)
155{
156 /* Calcul de la longueur paddée (PKCS#7) */
157 size_t padded_len = ((plain_len + 15) / 16) * 16;
158 if (padded_len > LEXACARE_MAX_PAYLOAD) {
159 ESP_LOGE(TAG, "Payload trop grand : %u octets (max %d)",
160 (unsigned)padded_len, LEXACARE_MAX_PAYLOAD);
161 return 0;
162 }
163
164 /* Buffer temporaire avec padding */
165 uint8_t padded[LEXACARE_MAX_PAYLOAD] = {0};
166 memcpy(padded, plain, plain_len);
167 uint8_t pad_byte = (uint8_t)(padded_len - plain_len);
168 for (size_t i = plain_len; i < padded_len; i++) {
169 padded[i] = pad_byte;
170 }
171
172 /* Génération de l'IV aléatoire */
173 esp_fill_random(iv, LEXACARE_AES_IV_LEN);
174
175 /* Chiffrement AES-128-CBC */
176 aes_compat_ctx_t aes_ctx;
177 aes_compat_init(&aes_ctx);
178
179 uint8_t iv_copy[LEXACARE_AES_IV_LEN];
180 memcpy(iv_copy, iv, LEXACARE_AES_IV_LEN); /* L'API AES modifie l'IV en place */
181
182 int ret = aes_compat_setkey(&aes_ctx, s_mesh.aes_key,
183 LEXACARE_AES_KEY_LEN * 8);
184 if (ret != 0) {
185 aes_compat_free(&aes_ctx);
186 ESP_LOGE(TAG, "AES setkey échoué : %d", ret);
187 return 0;
188 }
189
190 ret = aes_compat_crypt_cbc(&aes_ctx, AES_COMPAT_ENCRYPT,
191 padded_len, iv_copy, padded, cipher);
192 aes_compat_free(&aes_ctx);
193
194 if (ret != 0) {
195 ESP_LOGE(TAG, "AES CBC encrypt échoué : %d", ret);
196 return 0;
197 }
198
199 return padded_len;
200}

202/* ================================================================
203 * mesh_build_ai_frame (interne)
204 * @brief Construit une lexacare_frame_t depuis un ai_event_t.
205 *
206 * @param event Événement IA à encapsuler.
207 * @param frame Trame destination.
208 * @return true si la construction réussit.
209 * ================================================================ */
210static bool mesh_build_ai_frame(const ai_event_t *event,
211 lexacare_frame_t *frame)
212{
213 memset(frame, 0, sizeof(*frame));
214 frame->magic = LEXACARE_FRAME_MAGIC;
215 frame->type = FRAME_TYPE_AI_EVENT;
216 frame->hop_count = 0;
217 memcpy(frame->src_mac, s_mesh.own_mac, 6);
218
219 /* Chiffrement du payload */
220 size_t enc_len = aes_encrypt_payload((const uint8_t *)event,
221 sizeof(ai_event_t),
222 frame->payload,
223 frame->iv);
224 if (enc_len == 0) return false;
225 frame->payload_len = (uint16_t)enc_len;
226
227 /* CRC16 sur magic + type + hop + src_mac + iv + payload */
228 uint8_t crc_buf[2 + 1 + 1 + 6 + LEXACARE_AES_IV_LEN + LEXACARE_MAX_PAYLOAD];
229 size_t crc_len = 0;
230 memcpy(crc_buf + crc_len, &frame->magic, 2); crc_len += 2;
231 crc_buf[crc_len++] = frame->type;
232 crc_buf[crc_len++] = frame->hop_count;
233 memcpy(crc_buf + crc_len, frame->src_mac, 6); crc_len += 6;
234 memcpy(crc_buf + crc_len, frame->iv, LEXACARE_AES_IV_LEN);
235 crc_len += LEXACARE_AES_IV_LEN;
236 memcpy(crc_buf + crc_len, frame->payload, enc_len);
237 crc_len += enc_len;
238
239 frame->crc16 = crc16_ccitt(crc_buf, crc_len);
240 return true;
241}

303/* ================================================================
304 * mesh_load_aes_key (interne)
305 * @brief Charge la clé AES depuis NVS ou utilise la clé par défaut.
306 * ================================================================ */
307static void mesh_load_aes_key(void)
308{
309 nvs_handle_t nvs;
310 esp_err_t ret = nvs_open("lexacare", NVS_READONLY, &nvs);
311 if (ret == ESP_OK) {
312 size_t key_len = LEXACARE_AES_KEY_LEN;
313 ret = nvs_get_blob(nvs, "aes_key", s_mesh.aes_key, &key_len);
314 nvs_close(nvs);
315 if (ret == ESP_OK && key_len == LEXACARE_AES_KEY_LEN) {
316 ESP_LOGI(TAG, "Clé AES chargée depuis NVS");
317 return;
318 }
319 }
320
321 /* Clé par défaut (avertissement de sécurité) */
322 memcpy(s_mesh.aes_key, k_aes_default, LEXACARE_AES_KEY_LEN);
323 ESP_LOGW(TAG, "Clé AES par défaut utilisée — NE PAS utiliser en production");
324}

326/* ================================================================
327 * task_mesh_com (interne)
328 * @brief Tâche FreeRTOS de communication mesh — Core 0, priorité 8.
329 *
330 * @param pvParam Pointeur vers mesh_state_t.
331 * ================================================================ */
332static void task_mesh_com(void *pvParam)
333{
334 mesh_state_t *state = (mesh_state_t *)pvParam;
335 ESP_LOGI(TAG, "Task_Mesh_Com démarrée sur Core %d", xPortGetCoreID());
336
337 ai_event_t event;
338 lexacare_frame_t frame;
339
340 while (true) {
341 /* Attente d'un événement IA (portMAX_DELAY) */
342 if (xQueueReceive(state->sys_ctx->ai_to_mesh_queue,
343 &event, portMAX_DELAY) != pdTRUE) {
344 continue;
345 }
346
347 if (!state->has_parent && !state->sys_ctx->is_root_node) {
348 ESP_LOGW(TAG, "Pas de parent — événement non transmis");
349 continue;
350 }
351
352 /* Construction et chiffrement de la trame */
353 if (!mesh_build_ai_frame(&event, &frame)) {
354 ESP_LOGE(TAG, "Échec construction trame");
355 continue;
356 }
357
358 /* Envoi vers le parent (ou broadcast si ROOT) */
359 const uint8_t *dest = state->has_parent
360 ? state->parent_mac
361 : k_broadcast_mac;
362
363 esp_err_t send_ret = esp_now_send(dest,
364 (const uint8_t *)&frame,
365 sizeof(lexacare_frame_t));
366 if (send_ret != ESP_OK) {
367 ESP_LOGW(TAG, "esp_now_send échoué : %s", esp_err_to_name(send_ret));
368 }
369 }
370}
