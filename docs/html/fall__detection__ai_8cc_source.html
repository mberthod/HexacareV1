4 * @brief Moteur d'inférence de détection de chute — dual-mode ESP-DL / seuil.
5 *
6 * Ce fichier est compilé en C++ (requis pour l'API ESP-DL).
7 * L'interface externe est exposée via extern "C" pour la compatibilité
8 * avec main.c et les autres composants C.
9 *
10 * === MODE MODÈLE (CONFIG_AI_USE_ESPDL_MODEL=y) ===
11 * Charge /littlefs/fall_model.espdl via dl::Model::load().
12 * Construit un Tensor<int16_t> 8×32×1 depuis lidar_matrix_t.
13 * Exécute model->run(). Lit la sortie softmax [P_normal, P_chute].
14 * Si P_chute > CONFIG_AI_FALL_CONFIDENCE_THRESHOLD% → AI_CHUTE_DETECTEE.
15 *
16 * === MODE SEUIL (CONFIG_AI_USE_ESPDL_MODEL=n, défaut) ===
17 * Algorithme géométrique sur lidar_matrix_t :
18 * 1. Calcul de la distance moyenne de la zone inférieure (lignes 6-7).
19 * 2. Comparaison avec l'historique glissant (CONFIG_AI_HISTORY_LEN trames).
20 * 3. Si variation > CONFIG_AI_FALL_THRESHOLD_PERCENT% ET distance < 1500 mm
21 * → AI_CHUTE_DETECTEE (confiance proportionnelle à la variation).
22 *
23 * === RÈGLE DE PRIORITÉ MESH (contrainte stricte) ===
24 * AI_CHUTE_DETECTEE → xQueueSendToFront(ai_to_mesh_queue, ...)
25 * Autres états → xQueueSend(ai_to_mesh_queue, ...)
26 */

96/* ================================================================
97 * threshold_avg_lower_zone (interne)
98 * @brief Calcule la distance moyenne des lignes inférieures (6-7) de la matrice.
99 *
100 * Les lignes du bas correspondent à la zone sol — une distance faible
101 * et soudaine indique une personne allongée (chute potentielle).
102 *
103 * @param matrix Matrice LIDAR 8×32.
104 * @return Distance moyenne en mm (0 si toutes les zones invalides).
105 * ================================================================ */
106static float threshold_avg_lower_zone(const lidar_matrix_t *matrix)
107{
108 float sum = 0.0f;
109 int count = 0;
110
111 /* Lignes 6 et 7 = zone la plus basse du champ de vision */
112 for (int row = 6; row < LIDAR_ROWS; row++) {
113 for (int col = 0; col < LIDAR_COLS; col++) {
114 uint16_t d = matrix->data[row][col];
115 if (d > 0 && d < 5000) { /* Filtrer les mesures invalides */
116 sum += (float)d;
117 count++;
118 }
119 }
120 }
121
122 return (count > 0) ? (sum / (float)count) : 0.0f;
123}

125/* ================================================================
126 * threshold_detect_fall (interne)
127 * @brief Algorithme de détection par seuil sur l'historique glissant.
128 *
129 * @param ctx Contexte de la tâche (historique).
130 * @param current Distance moyenne actuelle de la zone inférieure.
131 * @param confidence Pointeur vers la confiance calculée (0–100).
132 * @return ai_event_state_e détecté.
133 * ================================================================ */
134static ai_event_state_e threshold_detect_fall(ai_task_ctx_t *ctx,
135 float current_mm,
136 uint8_t *confidence)
137{
138 /* Mise à jour de l'historique */
139 ctx->dist_history[ctx->history_head] = current_mm;
140 ctx->history_head = (ctx->history_head + 1) % CONFIG_AI_HISTORY_LEN;
141 if (ctx->history_head == 0) ctx->history_full = true;
142
143 int len = ctx->history_full ? CONFIG_AI_HISTORY_LEN : ctx->history_head;
144 if (len < 2) {
145 *confidence = 0;
146 return AI_NORMAL;
147 }
148
149 /* Référence = moyenne des (len-1) premières valeurs de l'historique */
150 float ref_sum = 0.0f;
151 for (int i = 0; i < len - 1; i++) {
152 int idx = (ctx->history_head - len + i + CONFIG_AI_HISTORY_LEN)
153 % CONFIG_AI_HISTORY_LEN;
154 ref_sum += ctx->dist_history[idx];
155 }
156 float ref_avg = ref_sum / (float)(len - 1);
157
158 if (ref_avg < 1.0f) {
159 *confidence = 0;
160 return AI_NORMAL;
161 }
162
163 /* Variation relative entre la mesure actuelle et la référence historique */
164 float variation_pct = ((ref_avg - current_mm) / ref_avg) * 100.0f;
165
166 ESP_LOGD(TAG, "Zone bas: %.0f mm, réf: %.0f mm, variation: %.1f%%",
167 current_mm, ref_avg, variation_pct);
168
169 /* Seuil de chute : distance faible ET variation significative */
170 float threshold = (float)CONFIG_AI_FALL_THRESHOLD_PERCENT;
171
172 if (variation_pct >= threshold && current_mm < (float)FALL_DISTANCE_MAX_MM) {
173 /* Confiance proportionnelle à la variation au-delà du seuil */
174 float excess = variation_pct - threshold;
175 float conf = 50.0f + (excess / threshold) * 50.0f;
176 *confidence = (uint8_t)std::min(conf, 100.0f);
177 return AI_CHUTE_DETECTEE;
178 }
179
180 if (variation_pct >= threshold * 0.5f) {
181 *confidence = (uint8_t)(variation_pct / threshold * 40.0f);
182 return AI_MOUVEMENT_ANORMAL;
183 }
184
185 *confidence = 0;
186 return AI_NORMAL;
187}

190/* ================================================================
191 * espdl_detect_fall (interne)
192 * @brief Inférence CNN via ESP-DL.
193 *
194 * @param ctx Contexte de la tâche (modèle chargé).
195 * @param matrix Matrice LIDAR 8×32.
196 * @param confidence Confiance retournée (0–100).
197 * @return État détecté.
198 * ================================================================ */
199static ai_event_state_e espdl_detect_fall(ai_task_ctx_t *ctx,
200 const lidar_matrix_t *matrix,
201 uint8_t *confidence)
202{
203 if (!ctx->model_loaded) {
204 *confidence = 0;
205 return AI_NORMAL;
206 }
207
208 /* Construction du tenseur d'entrée int16_t [1, 8, 32, 1]
209 * (Décommenter après intégration de l'ULD ESP-DL) :
210 *
211 * int16_t input_data[LIDAR_ROWS * LIDAR_COLS];
212 * for (int r = 0; r < LIDAR_ROWS; r++) {
213 * for (int c = 0; c < LIDAR_COLS; c++) {
214 * input_data[r * LIDAR_COLS + c] = (int16_t)matrix->data[r][c];
215 * }
216 * }
217 *
218 * dl::Tensor<int16_t> input;
219 * input.set_shape({1, LIDAR_ROWS, LIDAR_COLS, 1});
220 * input.set_element(input_data);
221 *
222 * ctx->model->run(&input);
223 *
224 * dl::Tensor<float> *output = ctx->model->get_output(0);
225 * float p_normal = output->get_element_ptr()[0];
226 * float p_fall = output->get_element_ptr()[1];
227 *
228 * *confidence = (uint8_t)(p_fall * 100.0f);
229 * if (*confidence >= CONFIG_AI_FALL_CONFIDENCE_THRESHOLD) {
230 * return AI_CHUTE_DETECTEE;
231 * }
232 * if (p_fall >= 0.3f) {
233 * return AI_MOUVEMENT_ANORMAL;
234 * }
235 */
236
237 *confidence = 0;
238 return AI_NORMAL;
239}

242/* ================================================================
243 * send_ai_event (interne)
244 * @brief Pousse un ai_event_t dans les queues réseau et diagnostic.
245 *
246 * Règle de priorité (contrainte stricte) :
247 * AI_CHUTE_DETECTEE → xQueueSendToFront (court-circuite la télémétrie)
248 * Autres états → xQueueSend (FIFO normale)
249 *
250 * @param ctx Contexte de la tâche.
251 * @param event Événement à envoyer.
252 * ================================================================ */
253static void send_ai_event(const ai_task_ctx_t *ctx, const ai_event_t *event)
254{
255 BaseType_t mesh_ret, diag_ret;
256
257 if (event->state == AI_CHUTE_DETECTEE) {
258 /* Priorité HAUTE : injection en tête de queue réseau */
259 mesh_ret = xQueueSendToFront(ctx->sys_ctx->ai_to_mesh_queue,
260 event, 0);
261 ESP_LOGW(TAG, "CHUTE DÉTECTÉE (confiance %u%%) — priorité mesh activée",
262 event->confidence);
263 } else {
264 /* Priorité normale : FIFO standard */
265 mesh_ret = xQueueSend(ctx->sys_ctx->ai_to_mesh_queue, event, 0);
266 }
267
268 /* Diagnostic PC : toujours FIFO */
269 diag_ret = xQueueSend(ctx->sys_ctx->diag_to_pc_queue, event, 0);
270
271 if (mesh_ret != pdTRUE) {
272 ESP_LOGD(TAG, "ai_to_mesh_queue pleine — événement perdu");
273 }
274 if (diag_ret != pdTRUE) {
275 ESP_LOGD(TAG, "diag_to_pc_queue pleine — événement perdu");
276 }
277}

279/* ================================================================
280 * task_ai_inference (interne)
281 * @brief Tâche FreeRTOS d'inférence — Core 1, priorité 9.
282 *
283 * @param pvParam Pointeur vers ai_task_ctx_t alloué par ai_engine_task_start.
284 * ================================================================ */
285static void task_ai_inference(void *pvParam)
286{
287 ai_task_ctx_t *ctx = static_cast<ai_task_ctx_t *>(pvParam);
288
289 /* Abonnement au TWDT */
290 ESP_ERROR_CHECK(esp_task_wdt_add(NULL));
291
292#ifdef CONFIG_AI_USE_ESPDL_MODEL
293 /* Chargement du modèle ESP-DL depuis LittleFS */
294 /* ctx->model = dl::Model::load("/littlefs/fall_model.espdl", MEM_AUTO); */
295 /* ctx->model_loaded = (ctx->model != nullptr); */
296 ctx->model_loaded = false; /* Passer à true après intégration de l'ULD */
297 if (!ctx->model_loaded) {
298 ESP_LOGE(TAG, "Modèle /littlefs/fall_model.espdl non trouvé — "
299 "basculement sur mode seuil de secours");
300 } else {
301 ESP_LOGI(TAG, "Modèle ESP-DL chargé (seuil confiance : %d%%)",
302 CONFIG_AI_FALL_CONFIDENCE_THRESHOLD);
303 }
304#else
305 ESP_LOGI(TAG, "Mode seuil actif (seuil variation : %d%%, historique : %d trames)",
306 CONFIG_AI_FALL_THRESHOLD_PERCENT, CONFIG_AI_HISTORY_LEN);
307#endif
308
309 ESP_LOGI(TAG, "Task_AI_Inference démarrée sur Core %d", xPortGetCoreID());
310
311 sensor_frame_t frame;
312
313 while (true) {
314 /* Reset watchdog même si aucune trame n'arrive encore.
315 * Important : au boot, la tâche d'acquisition peut démarrer après nous
316 * (ou être absente si LIDAR KO). Un blocage portMAX_DELAY déclenche le TWDT. */
317 ESP_ERROR_CHECK(esp_task_wdt_reset());
318
319 /* Attente d'une trame capteur avec timeout (permet de nourrir le TWDT) */
320 if (xQueueReceive(ctx->sys_ctx->sensor_to_ai_queue,
321 &frame, pdMS_TO_TICKS(500)) != pdTRUE) {
322 continue;
323 }
324
325 if (!frame.lidar_valid) {
326 continue; /* Ignorer les trames sans données LIDAR */
327 }
328
329 ai_event_t event = {
330 .state = AI_NORMAL,
331 .confidence = 0,
332 .timestamp_us = esp_timer_get_time(),
333 };
334
335#ifdef CONFIG_AI_USE_ESPDL_MODEL
336 if (ctx->model_loaded) {
337 event.state = espdl_detect_fall(ctx, &frame.lidar, &event.confidence);
338 } else {
339 /* Repli sur le mode seuil si le modèle n'est pas disponible */
340 float avg = threshold_avg_lower_zone(&frame.lidar);
341 event.state = threshold_detect_fall(ctx, avg, &event.confidence);
342 }
343#else
344 float avg = threshold_avg_lower_zone(&frame.lidar);
345 event.state = threshold_detect_fall(ctx, avg, &event.confidence);
346#endif
347
348 /* Envoi des événements non normaux */
349 if (event.state != AI_NORMAL) {
350 send_ai_event(ctx, &event);
351 }
352 }
353}

355/* ================================================================
356 * ai_engine_task_start (extern "C")
357 * @brief Alloue le contexte et crée Task_AI_Inference sur Core 1.
358 *
359 * @param ctx Pointeur vers le contexte système.
360 * @return ESP_OK si la tâche est créée avec succès.
361 * ================================================================ */
362extern "C" esp_err_t ai_engine_task_start(sys_context_t *ctx)
363{
364 ai_task_ctx_t *task_ctx = static_cast<ai_task_ctx_t *>(
365 calloc(1, sizeof(ai_task_ctx_t)));
366 if (!task_ctx) {
367 return ESP_ERR_NO_MEM;
368 }
369
370 task_ctx->sys_ctx = ctx;
371 task_ctx->history_head = 0;
372 task_ctx->history_full = false;
373 memset(task_ctx->dist_history, 0, sizeof(task_ctx->dist_history));
374
375 BaseType_t ret = xTaskCreatePinnedToCore(
376 task_ai_inference,
377 "Task_AI_Inference",
378 TASK_STACK_SIZE,
379 task_ctx,
380 TASK_PRIORITY,
381 nullptr,
382 TASK_CORE);
383
384 if (ret != pdPASS) {
385 free(task_ctx);
386 ESP_LOGE(TAG, "Échec création Task_AI_Inference");
387 return ESP_FAIL;
388 }
389
390 ESP_LOGI(TAG, "Task_AI_Inference créée (Core %d, priorité %d)",
391 TASK_CORE, TASK_PRIORITY);
392 return ESP_OK;
393}
