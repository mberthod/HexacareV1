1/**
2 * @file littlefs_manager.c
3 * @ingroup group_storage
4 * @brief Gestion du montage SPIFFS sur la partition "storage".
5 *
6 * Implémentation via esp_spiffs (built-in ESP-IDF v5.4).
7 * L'API publique reste identique — les fichiers POSIX (fopen/fclose/etc.)
8 * fonctionnent de la même façon avec SPIFFS qu'avec LittleFS.
9 *
10 * Note : SPIFFS ne supporte pas les répertoires. Tous les fichiers
11 * sont à la racine du point de montage (/spiffs/).
12 */

21/* ================================================================
22 * littlefs_manager_init
23 * @brief Monte la partition SPIFFS avec la configuration complète.
24 *
25 * Utilise esp_vfs_spiffs_conf_t (équivalent de la "root file
26 * configuration" de LittleFS) :
27 * - base_path = STORAGE_MOUNT_POINT ("/spiffs")
28 * - partition_label = STORAGE_PARTITION ("storage")
29 * - max_files = 8
30 * - format_if_mount_failed = true
31 *
32 * Après montage, l'espace est vérifié avec esp_spiffs_info().
33 *
34 * @return ESP_OK si le montage réussit, code d'erreur sinon.
35 * ================================================================ */
36esp_err_t littlefs_manager_init(void)
37{
38 /* Configuration complète du montage SPIFFS */
39 const esp_vfs_spiffs_conf_t conf = {
40 .base_path = STORAGE_MOUNT_POINT,
41 .partition_label = STORAGE_PARTITION,
42 .max_files = 8,
43 .format_if_mount_failed = true,
44 };
45
46 esp_err_t ret = esp_vfs_spiffs_register(&conf);
47 if (ret != ESP_OK) {
48 if (ret == ESP_FAIL) {
49 ESP_LOGE(TAG, "Échec du montage ou du formatage de SPIFFS");
50 } else if (ret == ESP_ERR_NOT_FOUND) {
51 ESP_LOGE(TAG, "Partition '%s' introuvable dans la table de partitions",
52 STORAGE_PARTITION);
53 } else {
54 ESP_LOGE(TAG, "Erreur montage SPIFFS : %s", esp_err_to_name(ret));
55 }
56 return ret;
57 }
58
59 /* Vérification post-montage de l'espace disponible */
60 size_t total_bytes = 0;
61 size_t used_bytes = 0;
62 ESP_ERROR_CHECK(esp_spiffs_info(STORAGE_PARTITION, &total_bytes, &used_bytes));
63
64 ESP_LOGI(TAG, "SPIFFS monté sur '%s' — %u Ko total, %u Ko utilisés, %u Ko libres",
65 STORAGE_MOUNT_POINT,
66 (unsigned)(total_bytes / 1024),
67 (unsigned)(used_bytes / 1024),
68 (unsigned)((total_bytes - used_bytes) / 1024));
69
70 return ESP_OK;
71}
