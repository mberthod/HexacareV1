33/* ================================================================
34 * mic_driver_init
35 * ================================================================ */
36esp_err_t mic_driver_init(void)
37{
38 /* Création du canal I2S RX */
39 const i2s_chan_config_t chan_cfg = {
40 .id = I2S_NUM_AUTO,
41 .role = I2S_ROLE_MASTER,
42 .dma_desc_num = 4,
43 .dma_frame_num = 256,
44 .auto_clear = true,
45 };
46
47 esp_err_t ret = i2s_new_channel(&chan_cfg, NULL, &s_rx_chan);
48 if (ret != ESP_OK) {
49 ESP_LOGE(TAG, "Création canal I2S échouée : %s", esp_err_to_name(ret));
50 return ret;
51 }
52
53 /* Configuration standard I2S */
54 i2s_std_config_t std_cfg = {
55 .clk_cfg = I2S_STD_CLK_DEFAULT_CONFIG(MIC_SAMPLE_RATE),
56 .slot_cfg = I2S_STD_MSB_SLOT_DEFAULT_CONFIG(
57 I2S_DATA_BIT_WIDTH_32BIT,
58 I2S_SLOT_MODE_STEREO), /* Stéréo pour lire LRCK */
59 .gpio_cfg = {
60 .mclk = I2S_GPIO_UNUSED,
61 .bclk = PIN_MIC_SCK,
62 .ws = PIN_MIC_WS,
63 .dout = I2S_GPIO_UNUSED,
64 .din = PIN_MIC_SD,
65 .invert_flags = {
66 .mclk_inv = false,
67 .bclk_inv = false,
68 .ws_inv = false,
69 },
70 },
71 };
72
73 ret = i2s_channel_init_std_mode(s_rx_chan, &std_cfg);
74 if (ret != ESP_OK) {
75 ESP_LOGE(TAG, "Init mode standard échoué : %s", esp_err_to_name(ret));
76 i2s_del_channel(s_rx_chan);
77 s_rx_chan = NULL;
78 return ret;
79 }
80
81 ret = i2s_channel_enable(s_rx_chan);
82 if (ret != ESP_OK) {
83 ESP_LOGE(TAG, "Activation canal I2S échouée : %s", esp_err_to_name(ret));
84 i2s_del_channel(s_rx_chan);
85 s_rx_chan = NULL;
86 return ret;
87 }
88
89 ESP_LOGI(TAG, "Microphone I2S initialisé (%d Hz, %d samples/burst)",
90 MIC_SAMPLE_RATE, MIC_BURST_SAMPLES);
91 return ESP_OK;
92}

94/* ================================================================
95 * mic_driver_read
96 * ================================================================ */
97esp_err_t mic_driver_read(uint32_t *rms, uint32_t *peak)
98{
99 if (!s_rx_chan) return ESP_ERR_INVALID_STATE;
100
101 size_t bytes_read = 0;
102 /* Lecture de MIC_BURST_SAMPLES × 2 canaux × 4 octets = burst stéréo */
103 const size_t read_size = MIC_BURST_SAMPLES * sizeof(int32_t) * 2;
104 static int32_t stereo_buf[MIC_BURST_SAMPLES * 2];
105
106 esp_err_t ret = i2s_channel_read(s_rx_chan, stereo_buf, read_size,
107 &bytes_read,
108 pdMS_TO_TICKS(MIC_READ_TIMEOUT_MS));
109 if (ret != ESP_OK) {
110 ESP_LOGW(TAG, "Lecture I2S échouée : %s", esp_err_to_name(ret));
111 return ret;
112 }
113
114 size_t n_samples = bytes_read / (sizeof(int32_t) * 2);
115 if (n_samples == 0) return ESP_ERR_TIMEOUT;
116
117 /* Extraction du canal gauche (pairs = index 0, 2, 4, ...) */
118 /* Les données MEMS INMP441 sont dans les 24 bits supérieurs (shift >> 8) */
119 uint64_t sum_sq = 0;
120 uint32_t peak_val = 0;
121
122 for (size_t i = 0; i < n_samples; i++) {
123 int32_t sample = stereo_buf[i * 2] >> 8; /* 24 bits significatifs */
124 uint32_t abs_val = (sample < 0) ? (uint32_t)(-sample) : (uint32_t)sample;
125 sum_sq += (uint64_t)abs_val * abs_val;
126 if (abs_val > peak_val) peak_val = abs_val;
127 }
128
129 *rms = (uint32_t)sqrtf((float)sum_sq / n_samples);
130 *peak = peak_val;
131
132 return ESP_OK;
133}
