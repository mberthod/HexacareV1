34/* ================================================================
35 * Constantes LIDAR
36 *
37 * Architecture :
38 * 4 capteurs VL53L8CX frontaux (8×8 zones chacun) concaténés
39 * horizontalement → matrice unifiée 8 lignes × 32 colonnes.
40 * Capteur N occupe les colonnes [N*8 .. N*8+7].
41 *
42 * 5e capteur (vertical/sol) : flux séparé, non inclus dans la matrice.
43 * ================================================================ */
44#define LIDAR_ROWS 8
45#define LIDAR_COLS 32
46#define LIDAR_NUM_FRONT 4 /**< Capteurs alimentant la matrice */
47#define LIDAR_NUM_TOTAL 5 /**< Total capteurs (4 frontaux + 1 sol) */
48#define LIDAR_ZONES_PER_SENSOR 64 /**< 8×8 = 64 zones par capteur */
49
50/* Adresses I2C (7 bits) allouées dynamiquement au démarrage */
51#define LIDAR_I2C_ADDR_DEFAULT 0x29 /**< Adresse par défaut VL53L8CX (7 bits) */
52#define LIDAR_I2C_ADDR_BASE 0x2A /**< Première adresse réassignée */

54/* ================================================================
55 * Constantes réseau et chiffrement AES-128
56 *
57 * Le payload de chaque trame lexacare_frame_t est chiffré en
58 * AES-128-CBC par le nœud émetteur (mbedTLS) indépendamment
59 * du LMK ESP-NOW — garantie de sécurité pour +1000 pairs.
60 * ================================================================ */
61#define LEXACARE_AES_KEY_LEN 16 /**< Clé AES-128 : 16 octets */
62#define LEXACARE_AES_IV_LEN 16 /**< IV AES-CBC : 16 octets */
63#define LEXACARE_FRAME_MAGIC 0xCA5E
64#define LEXACARE_MAX_PAYLOAD 128 /**< Payload chiffré (multiple de 16, ≤ 250 − header) */

87/* ================================================================

91 * Trames supportées :

103 bool presence; /**< true si une personne est détectée */
104 int64_t timestamp_us; /**< Horodatage µs */
105} radar_data_t;

129/* ================================================================
130 * ai_event_t
131 * @brief Événement généré par Task_AI_Inference.
132 *
133 * Règle de priorité (contrainte stricte) :
134 * AI_CHUTE_DETECTEE → xQueueSendToFront(ctx->ai_to_mesh_queue, ...)
135 * Autres états → xQueueSend(ctx->ai_to_mesh_queue, ...)

137 * Cette règle garantit que les alertes de chute court-circuitent

139 * ================================================================ */

143 int64_t timestamp_us;
144} ai_event_t;

146/* ================================================================
147 * lexacare_frame_t
148 * @brief Trame réseau ESP-NOW avec payload chiffré AES-128-CBC.
149 *
150 * Taille totale : 2+1+1+6+16+128+2+2 = 158 octets (< 250 max ESP-NOW).
151 *
152 * Séquence d'émission :
153 * 1. Sérialiser le payload en clair dans un buffer temporaire.
154 * 2. Générer un IV aléatoire (esp_fill_random).
155 * 3. Chiffrer avec mbedtls_aes_crypt_cbc (clé 16 o provisionnée en NVS).
156 * 4. Calculer CRC16 sur (magic + type + hop + src_mac + iv + payload).
157 * 5. Appeler esp_now_send().
158 * ================================================================ */
159typedef struct {
160 uint16_t magic; /**< LEXACARE_FRAME_MAGIC (0xCA5E) */
161 uint8_t type; /**< FRAME_TYPE_* */
162 uint8_t hop_count; /**< Nombre de sauts depuis la source */
163 uint8_t src_mac[6]; /**< MAC de l'émetteur original */
164 uint8_t iv[LEXACARE_AES_IV_LEN]; /**< IV aléatoire pour AES-128-CBC */
165 uint8_t payload[LEXACARE_MAX_PAYLOAD]; /**< Payload chiffré */
166 uint16_t payload_len; /**< Longueur du payload chiffré (octets) */
167 uint16_t crc16; /**< CRC16-CCITT sur header + iv + payload */
168} lexacare_frame_t;

177 uint16_t fragment_data_len; /**< Longueur des données utiles */
178 uint8_t data[OTA_FRAGMENT_DATA_LEN]; /**< Données du fragment */
179 bool is_last; /**< true si dernier fragment */
180} ota_fragment_payload_t;

182/* ================================================================
183 * sys_context_t
184 * @brief Contexte global injecté par pointeur dans toutes les tâches.
185 *
186 * Principe d'injection de dépendances :
187 * - Aucune variable globale directe.
188 * - Chaque tâche reçoit un pointeur const vers ce contexte.
189 * - Les handles I2C sont initialisés par hw_diag_run() et
190 * réutilisés par sensor_acq sans réinitialisation.
191 *
192 * Queues et dimensionnement :
193 * sensor_to_ai_queue : sensor_frame_t × 4 (profondeur conservatrice)
194 * ai_to_mesh_queue : ai_event_t × 8 (AI_CHUTE via xQueueSendToFront)
195 * diag_to_pc_queue : ai_event_t × 8
196 * log_queue : char* × 16 (pointeurs vers chaînes allouées)
197 *
198 * Capteurs environnementaux (I2C_NUM_1, SDA=10, SCL=9) :
199 * HDC1080 : temp/humidité (0x40)
200 * BME280 : temp/pression/humidité (0x76)
201 * ================================================================ */
202
