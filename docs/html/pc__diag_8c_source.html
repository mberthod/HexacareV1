2 * @file pc_diag.c
3 * @ingroup group_pc_diag
4 * @brief Interface PC USB-JTAG — Rapport JSON temps réel (Lexacare Studio).
5 *
6 * Émet deux types de trames JSON vers le PC via USB Serial/JTAG :
7 *
8 * TRAME SENSOR (toutes les 1 s — données temps réel) :
9 * {
10 * "type": "sensor",
11 * "lidar_matrix": [[...8 rangées × 32 colonnes (mm)...]],
12 * "radar": {
13 * "resp_phase": 0.5, "heart_phase": 0.1,
14 * "breath_rate": 16, "heart_rate": 72,
15 * "distance_mm": 1200, "presence": true
16 * },
17 * "ai_event": "AI_NORMAL",
18 * "confidence": 0,
19 * "hdc1080": { "temp_c": 23.5, "humidity_pct": 45.2 },
20 * "bme280": { "temp_c": 23.2, "pressure_hpa": 1013.2, "humidity_pct": 44.8 },
21 * "mic": { "rms": 1234, "peak": 4096 }
22 * }
23 *
24 * TRAME SYS (toutes les 5 s — métriques firmware) :
25 * {
26 * "type": "sys",
27 * "uptime_s": 1234,
28 * "cpu_load": [45, 80],
29 * "heap_free": 8700000,
30 * "psram_free": 8000000,
31 * "tasks_hwm": { "Task_Sensor_Acq": 2048, ... }
32 * }
33 *
34 * Commandes acceptées depuis stdin (JSON + '\n') :
35 * {"cmd":"get_diag"} → rapport JSON immédiat
36 * {"cmd":"download_logs"} → transfert /spiffs/app.log
37 * {"cmd":"clear_logs"} → suppression /spiffs/app.log
38 */

89/* ================================================================
90 * build_sensor_json (interne)
91 * @brief Construit la trame JSON "type:sensor" depuis le snapshot
92 * partagé de sys_context_t.
93 *
94 * @return Chaîne JSON allouée (libérer avec cJSON_free), ou NULL.
95 * ================================================================ */
96static char *build_sensor_json(pc_diag_ctx_t *ctx)
97{
98 sys_context_t *sys = ctx->sys_ctx;
99 cJSON *root = cJSON_CreateObject();
100 if (!root) return NULL;
101
102 cJSON_AddStringToObject(root, "type", "sensor");
103
104 /* ── Matrice LIDAR 8×32 ──────────────────────────────────────── */
105 sensor_frame_t snap = {0};
106 bool snap_valid = false;
107
108 if (sys->data_mutex &&
109 xSemaphoreTake(sys->data_mutex, pdMS_TO_TICKS(10)) == pdTRUE) {
110 snap = sys->latest_sensor;
111 snap_valid = sys->sensor_valid;
112 xSemaphoreGive(sys->data_mutex);
113 }
114
115 cJSON *matrix = cJSON_CreateArray();
116 for (int row = 0; row < LIDAR_ROWS; row++) {
117 cJSON *row_arr = cJSON_CreateArray();
118 for (int col = 0; col < LIDAR_COLS; col++) {
119 uint16_t d = snap_valid ? snap.lidar.data[row][col] : 0;
120 cJSON_AddItemToArray(row_arr, cJSON_CreateNumber(d));
121 }
122 cJSON_AddItemToArray(matrix, row_arr);
123 }
124 cJSON_AddItemToObject(root, "lidar_matrix", matrix);
125
126 /* ── Radar ───────────────────────────────────────────────────── */
127 cJSON *radar = cJSON_CreateObject();
128 if (snap_valid && snap.radar_valid) {
129 cJSON_AddNumberToObject(radar, "resp_phase", snap.radar.resp_phase);
130 cJSON_AddNumberToObject(radar, "heart_phase", snap.radar.heart_phase);
131 cJSON_AddNumberToObject(radar, "breath_rate", snap.radar.breath_rate_bpm);
132 cJSON_AddNumberToObject(radar, "heart_rate", snap.radar.heart_rate_bpm);
133 cJSON_AddNumberToObject(radar, "distance_mm", snap.radar.target_distance_mm);
134 cJSON_AddBoolToObject (radar, "presence", snap.radar.presence);
135 } else {
136 cJSON_AddNumberToObject(radar, "resp_phase", 0.0);
137 cJSON_AddNumberToObject(radar, "heart_phase", 0.0);
138 cJSON_AddNumberToObject(radar, "breath_rate", 0);
139 cJSON_AddNumberToObject(radar, "heart_rate", 0);
140 cJSON_AddNumberToObject(radar, "distance_mm", 0);
141 cJSON_AddBoolToObject (radar, "presence", false);
142 }
143 cJSON_AddItemToObject(root, "radar", radar);
144
145 /* ── Événement IA ────────────────────────────────────────────── */
146 /* Récupérer le dernier événement depuis la queue de diagnostic */
147 ai_event_t evt = ctx->last_event;
148 while (xQueueReceive(sys->diag_to_pc_queue, &evt, 0) == pdTRUE) {
149 ctx->last_event = evt;
150 ctx->has_event = true;
151 }
152 uint32_t ai_state = (ctx->has_event) ? ctx->last_event.state : 0;
153 const char *ai_str = (ai_state < AI_STATE_COUNT)
154 ? ai_event_strings[ai_state]
155 : "AI_NORMAL";
156 int confidence = (ctx->has_event) ? (int)ctx->last_event.confidence : 0;
157 cJSON_AddStringToObject(root, "ai_event", ai_str);
158 cJSON_AddNumberToObject(root, "confidence", confidence);
159
160 /* ── HDC1080 ─────────────────────────────────────────────────── */
161 cJSON *hdc = cJSON_CreateObject();
162 if (snap_valid) {
163 enviro_data_t env;
164 mic_data_t mic_data;
165 if (sys->data_mutex &&
166 xSemaphoreTake(sys->data_mutex, pdMS_TO_TICKS(5)) == pdTRUE) {
167 env = sys->enviro;
168 mic_data = sys->mic;
169 xSemaphoreGive(sys->data_mutex);
170 } else {
171 memset(&env, 0, sizeof(env));
172 memset(&mic_data, 0, sizeof(mic_data));
173 }
174
175 if (!isnan(env.temp_hdc_c))
176 cJSON_AddNumberToObject(hdc, "temp_c", (double)env.temp_hdc_c);
177 if (!isnan(env.humidity_hdc))
178 cJSON_AddNumberToObject(hdc, "humidity_pct", (double)env.humidity_hdc);
179
180 /* ── BME280 ─────────────────────────────────────────────────── */
181 cJSON *bme = cJSON_CreateObject();
182 if (!isnan(env.temp_bme_c))
183 cJSON_AddNumberToObject(bme, "temp_c", (double)env.temp_bme_c);
184 if (!isnan(env.pressure_hpa))
185 cJSON_AddNumberToObject(bme, "pressure_hpa", (double)env.pressure_hpa);
186 if (!isnan(env.humidity_bme))
187 cJSON_AddNumberToObject(bme, "humidity_pct", (double)env.humidity_bme);
188 cJSON_AddItemToObject(root, "bme280", bme);
189
190 /* ── Microphone ─────────────────────────────────────────────── */
191 cJSON *mic = cJSON_CreateObject();
192 cJSON_AddNumberToObject(mic, "rms", (double)mic_data.rms);
193 cJSON_AddNumberToObject(mic, "peak", (double)mic_data.peak);
194 cJSON_AddItemToObject(root, "mic", mic);
195 }
196 cJSON_AddItemToObject(root, "hdc1080", hdc);
197
198 char *json_str = cJSON_PrintUnformatted(root);
199 cJSON_Delete(root);
200 return json_str;
201}

203/* ================================================================
204 * build_sys_json (interne)
205 * @brief Construit la trame JSON "type:sys".
206 * ================================================================ */
207static char *build_sys_json(pc_diag_ctx_t *ctx)
208{
209 cJSON *root = cJSON_CreateObject();
210 if (!root) return NULL;
211
212 cJSON_AddStringToObject(root, "type", "sys");
213
214 int64_t uptime_us = esp_timer_get_time();
215 cJSON_AddNumberToObject(root, "uptime_s", (double)(uptime_us / 1000000LL));
216
217 /* cpu_load : non fourni nativement par ESP-IDF sans FreeRTOS runtime stats.
218 * On expose [0, 0] en attendant l'intégration de CONFIG_FREERTOS_GENERATE_RUN_TIME_STATS. */
219 cJSON *cpu_load = cJSON_CreateArray();
220 cJSON_AddItemToArray(cpu_load, cJSON_CreateNumber(0));
221 cJSON_AddItemToArray(cpu_load, cJSON_CreateNumber(0));
222 cJSON_AddItemToObject(root, "cpu_load", cpu_load);
223
224 cJSON_AddNumberToObject(root, "heap_free",
225 (double)heap_caps_get_free_size(MALLOC_CAP_INTERNAL));
226 cJSON_AddNumberToObject(root, "psram_free",
227 (double)heap_caps_get_free_size(MALLOC_CAP_SPIRAM));
228
229 /* Tâches FreeRTOS — High Water Mark */
230 TaskStatus_t task_statuses[MAX_TASKS];
231 UBaseType_t task_count = uxTaskGetSystemState(task_statuses, MAX_TASKS, NULL);
232
233 cJSON *tasks_hwm = cJSON_CreateObject();
234 for (UBaseType_t i = 0; i < task_count; i++) {
235 /* HWM en mots (4 octets) → convertir en octets */
236 uint32_t hwm_bytes = (uint32_t)task_statuses[i].usStackHighWaterMark
237 * sizeof(StackType_t);
238 cJSON_AddNumberToObject(tasks_hwm,
239 task_statuses[i].pcTaskName,
240 (double)hwm_bytes);
241 }
242 cJSON_AddItemToObject(root, "tasks_hwm", tasks_hwm);
243
244 char *json_str = cJSON_PrintUnformatted(root);
245 cJSON_Delete(root);
246 return json_str;
247}

249/* ================================================================

252static void cmd_download_logs(void)
253{
254 FILE *f = fopen(LOG_FILE_PATH, "r");
255 if (!f) {
256 printf("{\"status\":\"error\",\"msg\":\"Fichier log introuvable\"}\n");
257 fflush(stdout);
258 return;
259 }
260 fseek(f, 0, SEEK_END);
261 long file_size = ftell(f);
262 rewind(f);
263 printf("{\"status\":\"ok\",\"filename\":\"app.log\",\"size\":%ld}\n", file_size);
264 fflush(stdout);
265 uint8_t block[LOG_TRANSFER_BLOCK_SIZE];
266 size_t rb;
267 while ((rb = fread(block, 1, sizeof(block), f)) > 0) {
268 fwrite(block, 1, rb, stdout);
269 }
270 fclose(f);
271 printf("\n{\"status\":\"done\"}\n");
272 fflush(stdout);
273 ESP_LOGI(TAG, "Transfert log terminé (%ld octets)", file_size);
274}

288/* ================================================================
289 * process_command (interne)
290 * ================================================================ */
291static void process_command(pc_diag_ctx_t *ctx, const char *line)
292{
293 cJSON *cmd_json = cJSON_Parse(line);
294 if (!cmd_json) {
295 ESP_LOGW(TAG, "Commande JSON invalide : %s", line);
296 return;
297 }
298 const cJSON *cmd = cJSON_GetObjectItem(cmd_json, "cmd");
299 if (!cJSON_IsString(cmd)) {
300 cJSON_Delete(cmd_json);
301 return;
302 }
303 const char *cmd_str = cmd->valuestring;
304 ESP_LOGI(TAG, "Commande reçue : %s", cmd_str);
305
306 if (strcmp(cmd_str, "download_logs") == 0) {
307 cmd_download_logs();
308 } else if (strcmp(cmd_str, "clear_logs") == 0) {
309 cmd_clear_logs();
310 } else if (strcmp(cmd_str, "get_diag") == 0) {
311 char *json = build_sys_json(ctx);
312 if (json) { printf("%s\n", json); fflush(stdout); cJSON_free(json); }
313 } else {
314 printf("{\"status\":\"error\",\"msg\":\"Commande inconnue : %s\"}\n", cmd_str);
315 fflush(stdout);
316 }
317 cJSON_Delete(cmd_json);
318}

320/* ================================================================
321 * task_diag_pc (interne)
322 * @brief Tâche FreeRTOS — Core 0, priorité 3.
323 * ================================================================ */
324static void task_diag_pc(void *pvParam)
325{
326 pc_diag_ctx_t *ctx = (pc_diag_ctx_t *)pvParam;
327 ESP_LOGI(TAG, "Task_Diag_PC démarrée sur Core %d", xPortGetCoreID());
328
329 /* ── Stdin non-bloquant (IDF 6.0 / USB CDC) ──────────────────────
330 * fgets() peut bloquer indéfiniment sur USB JTAG en ESP-IDF 6.0.
331 * select() peut déclencher des allocations VFS récurrentes (esp_vfs_select)
332 * qui, sur certains builds, exposent des asserts TLSF.
333 * Stratégie : O_NONBLOCK + read() brut + parsing en mémoire. */
334 const int stdin_fd = fileno(stdin);
335 {
336 /* Tentative de mise en mode O_NONBLOCK (peut échouer sur certains VFS) */
337 int fl = fcntl(stdin_fd, F_GETFL, 0);
338 if (fl >= 0) fcntl(stdin_fd, F_SETFL, fl | O_NONBLOCK);
339 }
340
341 static char cmd_buf[CMD_BUF_SIZE]; /* statique pour économiser le stack */
342 static int cmd_len = 0;
343
344 TickType_t last_sensor = xTaskGetTickCount();
345 TickType_t last_sys = xTaskGetTickCount();
346
347 ESP_LOGI(TAG, "JSON démarré : sensor@%d s, sys@%d s",
348 SENSOR_INTERVAL_MS / 1000, SYS_INTERVAL_MS / 1000);
349
350 while (true) {
351 TickType_t now = xTaskGetTickCount();
352
353 /* ── Lecture commandes stdin (NON BLOQUANT) ───────────────────
354 * read() non bloquant : retourne -1/EAGAIN si rien à lire. */
355 {
356 char rx[64];
357 ssize_t r = read(stdin_fd, rx, sizeof(rx));
358 if (r > 0) {
359 for (ssize_t i = 0; i < r; i++) {
360 const char c = rx[i];
361 if (c == '\r') continue;
362 if (c == '\n' || cmd_len >= CMD_BUF_SIZE - 1) {
363 cmd_buf[cmd_len] = '\0';
364 if (cmd_len > 2) process_command(ctx, cmd_buf);
365 cmd_len = 0;
366 } else {
367 cmd_buf[cmd_len++] = c;
368 }
369 }
370 } else if (r < 0) {
371 if (!(errno == EAGAIN || errno == EWOULDBLOCK)) {
372 ESP_LOGW(TAG, "read(stdin) échoué : errno=%d", errno);
373 }
374 }
375 }
376
377 /* ── Rapport SENSOR (@SENSOR_INTERVAL_MS) ────────────────────
378 * Émis en JSON une ligne : {"type":"sensor","lidar_matrix":…} */
379 if ((now - last_sensor) >= pdMS_TO_TICKS(SENSOR_INTERVAL_MS)) {
380 last_sensor = now;
381 char *json = build_sensor_json(ctx);
382 if (json) {
383 printf("%s\n", json);
384 fflush(stdout);
385 cJSON_free(json);
386 } else {
387 ESP_LOGW(TAG, "build_sensor_json : allocation échouée (heap ?)");
388 }
389 }
390
391 /* ── Rapport SYS (@SYS_INTERVAL_MS) ─────────────────────────
392 * Émis en JSON une ligne : {"type":"sys","cpu_load":…} */
393 if ((now - last_sys) >= pdMS_TO_TICKS(SYS_INTERVAL_MS)) {
394 last_sys = now;
395 char *json = build_sys_json(ctx);
396 if (json) {
397 printf("%s\n", json);
398 fflush(stdout);
399 cJSON_free(json);
400 }
401 }
402
403 vTaskDelay(pdMS_TO_TICKS(50)); /* Cède le CPU — 50 ms = 20 itérations/s */
404 }
405}

410esp_err_t pc_diag_task_start(sys_context_t *ctx)
411{
412 pc_diag_ctx_t *task_ctx = calloc(1, sizeof(pc_diag_ctx_t));
413 if (!task_ctx) return ESP_ERR_NO_MEM;

428 ESP_LOGE(TAG, "Échec création Task_Diag_PC");

430 }
431
432 ESP_LOGI(TAG, "Task_Diag_PC créée (Core %d, prio %d, sensor@1Hz sys@5Hz)",
433 TASK_CORE, TASK_PRIORITY);
434 return ESP_OK;
