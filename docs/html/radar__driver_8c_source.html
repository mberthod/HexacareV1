6 * Protocole TinyFrame :

58/* ================================================================
59 * parse_tinyframe (interne)
60 * @brief Tente de décoder une trame TinyFrame dans le buffer.
61 *
62 * Si une trame valide est trouvée, met à jour s_parser.last_data.
63 * Consomme les octets traités du buffer (glissement).
64 *
65 * @return true si une trame a été décodée.
66 * ================================================================ */
67static bool parse_tinyframe(void)
68{
69 if (s_parser.len < TF_MIN_FRAME_LEN) return false;
70
71 /* Recherche du SOF */
72 size_t sof_pos = 0;
73 while (sof_pos < s_parser.len && s_parser.buf[sof_pos] != TF_SOF) {
74 sof_pos++;
75 }
76
77 if (sof_pos > 0) {
78 /* Supprimer les octets avant le SOF */
79 s_parser.len -= sof_pos;
80 memmove(s_parser.buf, s_parser.buf + sof_pos, s_parser.len);
81 }
82
83 if (s_parser.len < TF_MIN_FRAME_LEN) return false;
84
85 /* Décodage du header */
86 uint16_t type = ((uint16_t)s_parser.buf[1] << 8) | s_parser.buf[2];
87 uint8_t plen = s_parser.buf[3];
88
89 /* Vérification de la longueur totale */
90 size_t frame_len = TF_HEADER_LEN + plen + TF_FOOTER_LEN;
91 if (frame_len > sizeof(s_parser.buf)) {
92 /* Trame incohérente — resynchronisation */
93 s_parser.len = 0;
94 return false;
95 }
96
97 if (s_parser.len < frame_len) return false; /* Trame incomplète */
98
99 /* Vérification EOF */
100 if (s_parser.buf[TF_HEADER_LEN + plen] != TF_EOF) {
101 /* EOF invalide : sauter d'un octet et réessayer */
102 memmove(s_parser.buf, s_parser.buf + 1, --s_parser.len);
103 return false;
104 }
105
106 /* Extraction des données selon le type */
107 const uint8_t *payload = &s_parser.buf[TF_HEADER_LEN];
108 bool decoded = false;
109
110 switch (type) {
111 case TF_TYPE_BREATH:
112 if (plen >= 2) {
113 s_parser.last_data.breath_rate_bpm =
114 (uint16_t)(((uint16_t)payload[0] << 8) | payload[1]);
115 s_parser.last_data.presence = (s_parser.last_data.breath_rate_bpm > 0);
116 decoded = true;
117 ESP_LOGD(TAG, "Respiration : %u bpm",
118 s_parser.last_data.breath_rate_bpm);
119 }
120 break;
121
122 case TF_TYPE_HEART:
123 if (plen >= 2) {
124 s_parser.last_data.heart_rate_bpm =
125 (uint16_t)(((uint16_t)payload[0] << 8) | payload[1]);
126 decoded = true;
127 ESP_LOGD(TAG, "Cardio : %u bpm",
128 s_parser.last_data.heart_rate_bpm);
129 }
130 break;
131
132 case TF_TYPE_DISTANCE:
133 if (plen >= 2) {
134 s_parser.last_data.target_distance_mm =
135 (uint16_t)(((uint16_t)payload[0] << 8) | payload[1]);
136 decoded = true;
137 ESP_LOGD(TAG, "Distance : %u mm",
138 s_parser.last_data.target_distance_mm);
139 }
140 break;
141
142 default:
143 ESP_LOGV(TAG, "Type TinyFrame inconnu : 0x%04X", type);
144 break;
145 }
146
147 if (decoded) {
148 s_parser.last_data.timestamp_us = esp_timer_get_time();
149 s_parser.updated = true;
150 }
151
152 /* Consommer la trame du buffer */
153 memmove(s_parser.buf,
154 s_parser.buf + frame_len,
155 s_parser.len - frame_len);
156 s_parser.len -= frame_len;
157
158 return decoded;
159}

177/* ================================================================
178 * radar_driver_poll
179 * @brief Lit les octets UART disponibles et tente de décoder une trame.
180 *
181 * Non bloquant : timeout = 0 (retourne immédiatement si rien à lire).
182 * Appelé depuis la boucle de Task_Sensor_Acq.
183 *
184 * @param out Pointeur vers radar_data_t à mettre à jour si nouvelle trame.
185 * @return true si de nouvelles données ont été décodées.
186 * ================================================================ */
187bool radar_driver_poll(radar_data_t *out)
188{
189 /* Lecture non bloquante des octets disponibles */
190 size_t available = sizeof(s_parser.buf) - s_parser.len;
191 if (available > 0) {
192 int read = uart_read_bytes(UART_PORT,
193 s_parser.buf + s_parser.len,
194 available,
195 0 /* timeout = 0 ms, non bloquant */);
196 if (read > 0) {
197 s_parser.len += (size_t)read;
198 }
199 }
200
201 /* Tentative de décodage (peut trouver plusieurs trames) */
202 bool found = false;
203 while (parse_tinyframe()) {
204 found = true;
205 }
206
207 if (found && out) {
208 *out = s_parser.last_data;
209 s_parser.updated = false;
210 }
211
212 return found;
213}
