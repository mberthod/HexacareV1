17 * Inner function, not available outside this file. This function is used to
18 * wait for an answer from VL53L5 sensor.

21static uint8_t _vl53l8cx_poll_for_answer(
22 VL53L8CX_Configuration *p_dev,
23 uint16_t address,
24 uint8_t expected_value)
25{
26 uint8_t status = VL53L8CX_STATUS_OK;
27 uint8_t timeout = 0;
28
29 do {
30 status |= VL53L8CX_RdMulti(&(p_dev->platform),
31 address, p_dev->temp_buffer, 4);
32 status |= VL53L8CX_WaitMs(&(p_dev->platform), 10);
33
34 /* 2s timeout or FW error*/
35 if((timeout >= (uint8_t)200)
36 || (p_dev->temp_buffer[2] >= (uint8_t) 0x7f))
37 {
38 status |= VL53L8CX_MCU_ERROR;
39 break;
40 }
41 else
42 {
43 timeout++;
44 }
45 }while ((p_dev->temp_buffer[0x1]) != expected_value);
46
47 return status;
48}

55static uint8_t _vl53l8cx_program_output_config(
56 VL53L8CX_Configuration *p_dev)
57{
58 uint8_t resolution, status = VL53L8CX_STATUS_OK;
59 uint32_t i;
60 uint32_t header_config[2] = {0, 0};
61 union Block_header *bh_ptr;
62
63 status |= vl53l8cx_get_resolution(p_dev, &resolution);
64 p_dev->data_read_size = 0;
65
66 /* Enable mandatory output (meta and common data) */
67 uint32_t output_bh_enable[] = {
68 0x0001FFFFU,
69 0x00000000U,
70 0x00000000U,
71 0xC0000000U};
72
73 /* Send addresses of possible output */
74 uint32_t output[] ={
75 0x0000000DU,
76 0x54000040U,
77 0x9FD800C0U,
78 0x9FE40140U,
79 0x9FF80040U,
80 0x9FFC0404U,
81 0xA0FC0100U,
82 0xA10C0100U,
83 0xA11C00C0U,
84 0xA1280902U,
85 0xA2480040U,
86 0xA24C0081U,
87 0xA2540081U,
88 0xA25C0081U,
89 0xA2640081U,
90 0xA26C0084U,
91 0xA28C0082U};
92
93 /* Update data size */
94 for (i = 0; i < (uint32_t)(sizeof(output)/sizeof(uint32_t)); i++)
95 {
96 if ((output[i] == (uint8_t)0)
97 || ((output_bh_enable[i/(uint32_t)32]
98 &((uint32_t)1 << (i%(uint32_t)32))) == (uint32_t)0))
99 {
100 continue;
101 }
102
103 bh_ptr = (union Block_header *)&(output[i]);
104 if (((uint8_t)bh_ptr->type >= (uint8_t)0x1)
105 && ((uint8_t)bh_ptr->type < (uint8_t)0x0d))
106 {
107 if ((bh_ptr->idx >= (uint16_t)0x54d0)
108 && (bh_ptr->idx < (uint16_t)(0x54d0 + 960)))
109 {
110 bh_ptr->size = resolution;
111 }
112 else
113 {
114 bh_ptr->size = (uint8_t)(resolution
115 * (uint8_t)VL53L8CX_NB_TARGET_PER_ZONE);
116 }
117
118
119 p_dev->data_read_size += bh_ptr->type * bh_ptr->size;
120 }
121 else
122 {
123 p_dev->data_read_size += bh_ptr->size;
124 }
125
126 p_dev->data_read_size += (uint32_t)4;
127 }
128 p_dev->data_read_size += (uint32_t)24;
129
130 status |= vl53l8cx_dci_write_data(p_dev,
131 (uint8_t*)&(output),
132 VL53L8CX_DCI_OUTPUT_LIST, (uint16_t)sizeof(output));
133
134 header_config[0] = p_dev->data_read_size;
135 header_config[1] = i + (uint32_t)1;
136
137 status |= vl53l8cx_dci_write_data(p_dev, (uint8_t*)&(header_config),
138 VL53L8CX_DCI_OUTPUT_CONFIG,
139 (uint16_t)sizeof(header_config));
140 status |= vl53l8cx_dci_write_data(p_dev, (uint8_t*)&(output_bh_enable),
141 VL53L8CX_DCI_OUTPUT_ENABLES,
142 (uint16_t)sizeof(output_bh_enable));
143
144 return status;
145}

147uint8_t vl53l8cx_calibrate_xtalk(
148 VL53L8CX_Configuration *p_dev,
149 uint16_t reflectance_percent,
150 uint8_t nb_samples,
151 uint16_t distance_mm)
152{
153 uint16_t timeout = 0;
154 uint8_t cmd[] = {0x00, 0x03, 0x00, 0x00};
155 uint8_t footer[] = {0x00, 0x00, 0x00, 0x0F, 0x00, 0x01, 0x03, 0x04};
156 uint8_t continue_loop = 1, status = VL53L8CX_STATUS_OK;
157
158 uint8_t resolution, frequency, target_order, sharp_prct, ranging_mode;
159 uint32_t integration_time_ms, xtalk_margin;
160
161 uint16_t reflectance = reflectance_percent;
162 uint8_t samples = nb_samples;
163 uint16_t distance = distance_mm;
164 uint8_t *default_xtalk_ptr;
165
166 /* Get initial configuration */
167 status |= vl53l8cx_get_resolution(p_dev, &resolution);
168 status |= vl53l8cx_get_ranging_frequency_hz(p_dev, &frequency);
169 status |= vl53l8cx_get_integration_time_ms(p_dev, &integration_time_ms);
170 status |= vl53l8cx_get_sharpener_percent(p_dev, &sharp_prct);
171 status |= vl53l8cx_get_target_order(p_dev, &target_order);
172 status |= vl53l8cx_get_xtalk_margin(p_dev, &xtalk_margin);
173 status |= vl53l8cx_get_ranging_mode(p_dev, &ranging_mode);
174
175 /* Check input arguments validity */
176 if(((reflectance < (uint16_t)1) || (reflectance > (uint16_t)99))
177 || ((distance < (uint16_t)600) || (distance > (uint16_t)3000))
178 || ((samples < (uint8_t)1) || (samples > (uint8_t)16)))
179 {
180 status |= VL53L8CX_STATUS_INVALID_PARAM;
181 }
182 else
183 {
184 status |= vl53l8cx_set_resolution(p_dev,
185 VL53L8CX_RESOLUTION_8X8);
186
187 /* Send Xtalk calibration buffer */
188 (void)memcpy(p_dev->temp_buffer, VL53L8CX_CALIBRATE_XTALK,
189 sizeof(VL53L8CX_CALIBRATE_XTALK));
190 status |= VL53L8CX_WrMulti(&(p_dev->platform), 0x2c28,
191 p_dev->temp_buffer,
192 (uint16_t)sizeof(VL53L8CX_CALIBRATE_XTALK));
193 status |= _vl53l8cx_poll_for_answer(p_dev,
194 VL53L8CX_UI_CMD_STATUS, 0x3);
195
196 /* Format input argument */
197 reflectance = reflectance * (uint16_t)16;
198 distance = distance * (uint16_t)4;
199
200 /* Update required fields */
201 status |= vl53l8cx_dci_replace_data(p_dev, p_dev->temp_buffer,
202 VL53L8CX_DCI_CAL_CFG, 8,
203 (uint8_t*)&distance, 2, 0x00);
204
205 status |= vl53l8cx_dci_replace_data(p_dev, p_dev->temp_buffer,
206 VL53L8CX_DCI_CAL_CFG, 8,
207 (uint8_t*)&reflectance, 2,0x02);
208
209 status |= vl53l8cx_dci_replace_data(p_dev, p_dev->temp_buffer,
210 VL53L8CX_DCI_CAL_CFG, 8,
211 (uint8_t*)&samples, 1, 0x04);
212
213 /* Program output for Xtalk calibration */
214 status |= _vl53l8cx_program_output_config(p_dev);
215
216 /* Start ranging session */
217 status |= VL53L8CX_WrMulti(&(p_dev->platform),
218 VL53L8CX_UI_CMD_END - (uint16_t)(4 - 1),
219 (uint8_t*)cmd, sizeof(cmd));
220 status |= _vl53l8cx_poll_for_answer(p_dev,
221 VL53L8CX_UI_CMD_STATUS, 0x3);
222
223 /* Wait for end of calibration */
224 do {
225 status |= VL53L8CX_RdMulti(&(p_dev->platform),
226 0x0, p_dev->temp_buffer, 4);
227 if(p_dev->temp_buffer[0] != VL53L8CX_STATUS_ERROR)
228 {
229 /* Coverglass too good for Xtalk calibration */
230 if((p_dev->temp_buffer[2] >= (uint8_t)0x7f) &&
231 (((uint16_t)(p_dev->temp_buffer[3] &
232 (uint16_t)0x80) >> 7) == (uint16_t)1))
233 {
234 default_xtalk_ptr = p_dev->default_xtalk;
235 (void)memcpy(p_dev->xtalk_data,
236 default_xtalk_ptr,
237 sizeof(p_dev->xtalk_data));
238 status |= VL53L8CX_STATUS_XTALK_FAILED;
239 }
240 continue_loop = (uint8_t)0;
241 }
242 else if(timeout >= (uint16_t)400)
243 {
244 status |= VL53L8CX_STATUS_ERROR;
245 continue_loop = (uint8_t)0;
246 }
247 else
248 {
249 timeout++;
250 status |= VL53L8CX_WaitMs(&(p_dev->platform), 50);
251 }
252
253 }while (continue_loop == (uint8_t)1);
254 }
255
256 /* Save Xtalk data into the Xtalk buffer */
257 (void)memcpy(p_dev->temp_buffer, VL53L8CX_GET_XTALK_CMD,
258 sizeof(VL53L8CX_GET_XTALK_CMD));
259 status |= VL53L8CX_WrMulti(&(p_dev->platform), 0x2fb8,
260 p_dev->temp_buffer,
261 (uint16_t)sizeof(VL53L8CX_GET_XTALK_CMD));
262 status |= _vl53l8cx_poll_for_answer(p_dev,VL53L8CX_UI_CMD_STATUS, 0x03);
263 status |= VL53L8CX_RdMulti(&(p_dev->platform), VL53L8CX_UI_CMD_START,
264 p_dev->temp_buffer,
265 VL53L8CX_XTALK_BUFFER_SIZE + (uint16_t)4);
266
267 (void)memcpy(&(p_dev->xtalk_data[0]), &(p_dev->temp_buffer[8]),
268 VL53L8CX_XTALK_BUFFER_SIZE - (uint16_t)8);
269 (void)memcpy(&(p_dev->xtalk_data[VL53L8CX_XTALK_BUFFER_SIZE
270 - (uint16_t)8]), footer, sizeof(footer));
271
272 /* Reset default buffer */
273 status |= VL53L8CX_WrMulti(&(p_dev->platform), 0x2c34,
274 p_dev->default_configuration,
275 VL53L8CX_CONFIGURATION_SIZE);
276 status |= _vl53l8cx_poll_for_answer(p_dev,VL53L8CX_UI_CMD_STATUS, 0x03);
277
278 /* Reset initial configuration */
279 status |= vl53l8cx_set_resolution(p_dev, resolution);
280 status |= vl53l8cx_set_ranging_frequency_hz(p_dev, frequency);
281 status |= vl53l8cx_set_integration_time_ms(p_dev, integration_time_ms);
282 status |= vl53l8cx_set_sharpener_percent(p_dev, sharp_prct);
283 status |= vl53l8cx_set_target_order(p_dev, target_order);
284 status |= vl53l8cx_set_xtalk_margin(p_dev, xtalk_margin);
285 status |= vl53l8cx_set_ranging_mode(p_dev, ranging_mode);
286
287 return status;
288}

291 VL53L8CX_Configuration *p_dev,

293{

301 sizeof(VL53L8CX_GET_XTALK_CMD));
302 status |= VL53L8CX_WrMulti(&(p_dev->platform), 0x2fb8,
303 p_dev->temp_buffer, sizeof(VL53L8CX_GET_XTALK_CMD));

305 status |= VL53L8CX_RdMulti(&(p_dev->platform), VL53L8CX_UI_CMD_START,
306 p_dev->temp_buffer,
307 VL53L8CX_XTALK_BUFFER_SIZE + (uint16_t)4);
308
309 (void)memcpy(&(p_xtalk_data[0]), &(p_dev->temp_buffer[8]),
310 VL53L8CX_XTALK_BUFFER_SIZE-(uint16_t)8);
311 (void)memcpy(&(p_xtalk_data[VL53L8CX_XTALK_BUFFER_SIZE - (uint16_t)8]),

313

315

332uint8_t vl53l8cx_get_xtalk_margin(
333 VL53L8CX_Configuration *p_dev,
334 uint32_t *p_xtalk_margin)
335{
336 uint8_t status = VL53L8CX_STATUS_OK;
337
338 status |= vl53l8cx_dci_read_data(p_dev, (uint8_t*)p_dev->temp_buffer,
339 VL53L8CX_DCI_XTALK_CFG, 16);
340
341 (void)memcpy(p_xtalk_margin, p_dev->temp_buffer, 4);
342 *p_xtalk_margin = *p_xtalk_margin/(uint32_t)2048;
343
344 return status;

365
