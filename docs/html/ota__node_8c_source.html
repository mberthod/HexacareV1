1/**
2 * @file ota_node.c
3 * @ingroup group_mesh
4 * @brief OTA nœud-à-nœud via ESP-NOW — fragments 200 octets.
5 *
6 * Protocole :
7 * 1. Réception d'un fragment OTA_FRAGMENT déchiffré (ota_fragment_payload_t).
8 * 2. Premier fragment (offset == 0) : appel esp_ota_begin().
9 * 3. esp_ota_write() avec les données du fragment.
10 * 4. Retransmission immédiate du fragment brut (chiffré) aux enfants.
11 * 5. Dernier fragment (is_last == true) :
12 * esp_ota_end() → vérification MD5 → esp_ota_set_boot_partition()
13 * → log_write("OTA terminé") → esp_restart().
14 */

39/* ================================================================
40 * ota_node_init
41 * @brief Initialise l'état OTA (réinitialise si une OTA était en cours).
42 *
43 * @return ESP_OK toujours.
44 * ================================================================ */
45esp_err_t ota_node_init(void)
46{
47 memset(&s_ota, 0, sizeof(s_ota));
48
49 /* Identification de la partition de mise à jour */
50 s_ota.ota_partition = esp_ota_get_next_update_partition(NULL);
51 if (!s_ota.ota_partition) {
52 ESP_LOGW(TAG, "Aucune partition OTA disponible");
53 } else {
54 ESP_LOGI(TAG, "Partition OTA cible : %s (offset 0x%08lX, taille %lu Ko)",
55 s_ota.ota_partition->label,
56 s_ota.ota_partition->address,
57 s_ota.ota_partition->size / 1024);
58 }
59
60 return ESP_OK;
61}

63/* ================================================================
64 * ota_node_process_fragment
65 * @brief Traite un fragment OTA et le retransmet aux enfants.
66 *
67 * @param frag Payload du fragment (données en clair).
68 * @param children_mac Tableau des MACs des enfants (peut être NULL).
69 * @param child_count Nombre d'enfants.
70 * @param raw_frame Trame brute chiffrée à retransmettre.
71 * @param frame_len Longueur de la trame brute.
72 * @return ESP_OK si le traitement réussit.
73 * ================================================================ */
74esp_err_t ota_node_process_fragment(const ota_fragment_payload_t *frag,
75 const uint8_t (*children_mac)[6],
76 int child_count,
77 const uint8_t *raw_frame,
78 size_t frame_len)
79{
80 if (!frag) return ESP_ERR_INVALID_ARG;
81 if (!s_ota.ota_partition) {
82 ESP_LOGE(TAG, "Pas de partition OTA — fragment ignoré");
83 return ESP_FAIL;
84 }
85
86 /* Premier fragment : initialisation de l'OTA */
87 if (frag->fragment_offset == 0) {
88 if (s_ota.in_progress) {
89 ESP_LOGW(TAG, "Nouvelle OTA démarrée — abandon de l'OTA précédente");
90 esp_ota_abort(s_ota.ota_handle);
91 }
92
93 ESP_ERROR_CHECK(esp_ota_begin(s_ota.ota_partition,
94 OTA_WITH_SEQUENTIAL_WRITES,
95 &s_ota.ota_handle));
96 s_ota.in_progress = true;
97 s_ota.bytes_written = 0;
98 s_ota.firmware_size = frag->firmware_size;
99 ESP_LOGI(TAG, "OTA démarrée — firmware attendu : %lu octets",
100 frag->firmware_size);
101 }
102
103 if (!s_ota.in_progress) {
104 ESP_LOGW(TAG, "Fragment OTA reçu hors séquence — ignoré");
105 return ESP_FAIL;
106 }
107
108 /* Vérification de la cohérence de l'offset */
109 if (frag->fragment_offset != s_ota.bytes_written) {
110 ESP_LOGW(TAG, "Fragment hors ordre : attendu %lu, reçu %lu",
111 s_ota.bytes_written, frag->fragment_offset);
112 return ESP_FAIL;
113 }
114
115 /* Écriture du fragment dans la partition OTA */
116 ESP_ERROR_CHECK(esp_ota_write(s_ota.ota_handle,
117 frag->data,
118 frag->fragment_data_len));
119 s_ota.bytes_written += frag->fragment_data_len;
120
121 ESP_LOGD(TAG, "OTA : %lu / %lu octets écrits (%.1f%%)",
122 s_ota.bytes_written, s_ota.firmware_size,
123 (float)s_ota.bytes_written / (float)s_ota.firmware_size * 100.0f);
124
125 /* Retransmission immédiate aux enfants (propagation du fragment) */
126 if (raw_frame && frame_len > 0 && child_count > 0) {
127 for (int i = 0; i < child_count; i++) {
128 esp_err_t tx_ret = esp_now_send(children_mac[i],
129 raw_frame, frame_len);
130 if (tx_ret != ESP_OK) {
131 ESP_LOGW(TAG, "Retransmission OTA enfant %d échouée : %s",
132 i, esp_err_to_name(tx_ret));
133 }
134 }
135 ESP_LOGD(TAG, "Fragment retransmis à %d enfant(s)", child_count);
136 }
137
138 /* Dernier fragment : finalisation */
139 if (frag->is_last) {
140 ESP_LOGI(TAG, "Dernier fragment reçu — finalisation OTA");
141
142 esp_err_t end_ret = esp_ota_end(s_ota.ota_handle);
143 if (end_ret != ESP_OK) {
144 ESP_LOGE(TAG, "esp_ota_end échoué : %s", esp_err_to_name(end_ret));
145 s_ota.in_progress = false;
146 return end_ret;
147 }
148
149 esp_err_t boot_ret = esp_ota_set_boot_partition(s_ota.ota_partition);
150 if (boot_ret != ESP_OK) {
151 ESP_LOGE(TAG, "esp_ota_set_boot_partition échoué : %s",
152 esp_err_to_name(boot_ret));
153 s_ota.in_progress = false;
154 return boot_ret;
155 }
156
157 ESP_LOGI(TAG, "OTA terminée — redémarrage dans 1 s");
158 s_ota.in_progress = false;
159
160 /* Délai court pour laisser les logs s'afficher */
161 vTaskDelay(pdMS_TO_TICKS(1000));
162 esp_restart();
163 }
164
165 return ESP_OK;
166}
