1/**
2 * @file pins_config.h
3 * @brief Définition des broches GPIO et du mapping PCA9555 pour LexaCare V1.
4 *
5 * Architecture hardware ESP32-S3-WROOM-2 :
6 *
7 * SPI2 — 4 LIDARs VL53L8CX (NCS individuels) + SYNC
8 * UART — Radar HLK-LD6002 (RX=43, TX=44)
9 * I2C0 — PCA9555D @0x20 (SDA=11, SCL=12) : LPn LIDAR + alim sous-systèmes
10 * I2C1 — MLX90640 (SDA=10, SCL=9) — réservé, driver non implémenté
11 * I2S — Microphone MEMS numérique (WS=6, SD=7, SCK=8)
12 *
13 * PCA9555D Port 0 :
14 * IO0.0 → DIST_SHUTDOWN : shutdown VL53L0X (TOF simple, I2C MLX bus)
15 * IO0.1 → GPIO radar LD6002 — Hi-Z (non utilisé)
16 * IO0.2 → GPIO radar LD6002 — Hi-Z (non utilisé)
17 * IO0.3 → LPn_4 : LPn LIDAR 4 (VL53L8CX)
18 * IO0.4 → LPn_3 : LPn LIDAR 3 (VL53L8CX)
19 * IO0.5 → LPn_2 : LPn LIDAR 2 (VL53L8CX)
20 * IO0.6 → LPn_1 : LPn LIDAR 1 (VL53L8CX)
21 * IO0.7 → RADAR_LPN : LPn LIDAR carte radar (option, non connecté)
22 *
23 * PCA9555D Port 1 :
24 * IO1.0–IO1.3 → non utilisés (Hi-Z)
25 * IO1.4 → POWER_FAN : alimentation ventilateur
26 * IO1.5 → POWER_RADAR : alimentation carte radar
27 * IO1.6 → POWER_MIC : alimentation microphone
28 * IO1.7 → POWER_MLX : alimentation capteur MLX
29 *
30 * IMPORTANT : vérifier ces assignations contre le schéma électrique final
31 * avant tout flashage sur le matériel de production.
32 */

57/* ================================================================
58 * Bus SPI LIDAR (SPI2_HOST) — 4 capteurs VL53L8CX
59 *
60 * Numérotation hardware vs code :
61 * NCS0 (GPIO1) → LIDAR 1 (LPn_1 = PCA9555 IO0.6)
62 * NCS1 (GPIO2) → LIDAR 2 (LPn_2 = PCA9555 IO0.5)
63 * NCS2 (GPIO42) → LIDAR 3 (LPn_3 = PCA9555 IO0.4)
64 * NCS3 (GPIO41) → LIDAR 4 (LPn_4 = PCA9555 IO0.3)
65 *
66 * Mode SPI 3 (CPOL=1, CPHA=1) — max 10 MHz.
67 * SYNC (GPIO14) : synchronisation multi-capteurs.
68 * ================================================================ */
69#define LIDAR_SPI_HOST SPI2_HOST
70#define PIN_LIDAR_CLK 17
71#define PIN_LIDAR_MOSI 18
72#define PIN_LIDAR_MISO 21
73#define PIN_LIDAR_SYNC 14

75#define PIN_LIDAR_NCS0 1 /**< NCS LIDAR 1 */

91/* --- Port 0 --- */
92/* IO0.0 : DIST_SHUTDOWN — shutdown VL53L0X (TOF simple sur bus I2C MLX) */
93#define PCA9555_BIT_DIST_SHUTDOWN (1u << 0) /**< IO0.0 — actif = TOF en veille */
94
95/* IO0.1, IO0.2 : GPIO radar LD6002 — laissés en entrée (Hi-Z) */
96/* (pas de define actif — configurés en entrée dans pca9555_init) */
97
98/* IO0.3 – IO0.6 : LPn VL53L8CX (actif bas = reset, haut = opérationnel) */
99#define PCA9555_BIT_LPN4 (1u << 3) /**< IO0.3 → LPn_4 LIDAR 4 (NCS3/GPIO41) */
100#define PCA9555_BIT_LPN3 (1u << 4) /**< IO0.4 → LPn_3 LIDAR 3 (NCS2/GPIO42) */
101#define PCA9555_BIT_LPN2 (1u << 5) /**< IO0.5 → LPn_2 LIDAR 2 (NCS1/GPIO2) */
102#define PCA9555_BIT_LPN1 (1u << 6) /**< IO0.6 → LPn_1 LIDAR 1 (NCS0/GPIO1) */
103
104/* IO0.7 : LPn LIDAR carte radar (option — non connecté actuellement) */
105#define PCA9555_BIT_RADAR_LPN (1u << 7) /**< IO0.7 → LPn LIDAR radar (réservé) */
106
107/* Masque de config Port 0 :
108 * IO0.0 = sortie (DIST_SHUTDOWN)
109 * IO0.1, IO0.2 = entrée Hi-Z (radar GPIO non utilisés)
110 * IO0.3 – IO0.7 = sorties (LPn + RADAR_LPN)
111 * Config register : 0 = sortie, 1 = entrée → 0b00000110 = 0x06 */
112#define PCA9555_CFG0_MASK 0x06u
113
114/* --- Port 1 --- */
115/* IO1.0 – IO1.3 : non utilisés — laissés en entrée (Hi-Z) */
116/* IO1.4 – IO1.7 : alimentations sous-systèmes */
117#define PCA9555_BIT_PWR_FAN (1u << 4) /**< IO1.4 → Alimentation ventilateur */
118#define PCA9555_BIT_PWR_RADAR (1u << 5) /**< IO1.5 → Alimentation carte radar */
119#define PCA9555_BIT_PWR_MIC (1u << 6) /**< IO1.6 → Alimentation microphone */
120#define PCA9555_BIT_PWR_MLX (1u << 7) /**< IO1.7 → Alimentation MLX90640 */
121
122/* Masque de config Port 1 :
123 * IO1.0–IO1.3 = entrée Hi-Z, IO1.4–IO1.7 = sorties → 0b00001111 = 0x0F */
124#define PCA9555_CFG1_MASK 0x0Fu
