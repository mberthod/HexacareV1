68static esp_err_t read_calibration(void)
69{
70 uint8_t c[26] = {0};
71 esp_err_t ret = read_regs(REG_CALIB_00, c, 24);
72 if (ret != ESP_OK) return ret;
73
74 s_calib.T1 = (uint16_t)(c[1] << 8 | c[0]);
75 s_calib.T2 = (int16_t) (c[3] << 8 | c[2]);
76 s_calib.T3 = (int16_t) (c[5] << 8 | c[4]);
77 s_calib.P1 = (uint16_t)(c[7] << 8 | c[6]);
78 s_calib.P2 = (int16_t) (c[9] << 8 | c[8]);
79 s_calib.P3 = (int16_t) (c[11] << 8 | c[10]);
80 s_calib.P4 = (int16_t) (c[13] << 8 | c[12]);
81 s_calib.P5 = (int16_t) (c[15] << 8 | c[14]);
82 s_calib.P6 = (int16_t) (c[17] << 8 | c[16]);
83 s_calib.P7 = (int16_t) (c[19] << 8 | c[18]);
84 s_calib.P8 = (int16_t) (c[21] << 8 | c[20]);
85 s_calib.P9 = (int16_t) (c[23] << 8 | c[22]);
86
87 ret = read_regs(REG_CALIB_H1, c, 1);
88 if (ret != ESP_OK) return ret;
89 s_calib.H1 = c[0];
90
91 uint8_t h[7] = {0};
92 ret = read_regs(REG_CALIB_H2, h, 7);
93 if (ret != ESP_OK) return ret;
94 s_calib.H2 = (int16_t) (h[1] << 8 | h[0]);
95 s_calib.H3 = h[2];
96 s_calib.H4 = (int16_t) ((h[3] << 4) | (h[4] & 0x0F));
97 s_calib.H5 = (int16_t) ((h[5] << 4) | (h[4] >> 4));
98 s_calib.H6 = (int8_t) h[6];
99
100 return ESP_OK;
101}

107static float compensate_temperature(int32_t raw_t)
108{
109 int32_t var1 = ((((raw_t >> 3) - ((int32_t)s_calib.T1 << 1))) *
110 (int32_t)s_calib.T2) >> 11;
111 int32_t var2 = (((((raw_t >> 4) - (int32_t)s_calib.T1) *
112 ((raw_t >> 4) - (int32_t)s_calib.T1)) >> 12) *
113 (int32_t)s_calib.T3) >> 14;
114 s_t_fine = var1 + var2;
115 return (float)((s_t_fine * 5 + 128) >> 8) / 100.0f;
116}
117
118static float compensate_pressure(int32_t raw_p)
119{
120 int64_t var1 = (int64_t)s_t_fine - 128000;
121 int64_t var2 = var1 * var1 * (int64_t)s_calib.P6;
122 var2 += (var1 * (int64_t)s_calib.P5) << 17;
123 var2 += ((int64_t)s_calib.P4) << 35;
124 var1 = ((var1 * var1 * (int64_t)s_calib.P3) >> 8) +
125 ((var1 * (int64_t)s_calib.P2) << 12);
126 var1 = (((int64_t)1 << 47) + var1) * (int64_t)s_calib.P1 >> 33;
127 if (var1 == 0) return 0.0f;
128 int64_t p = 1048576 - raw_p;
129 p = (((p << 31) - var2) * 3125) / var1;
130 var1 = ((int64_t)s_calib.P9 * (p >> 13) * (p >> 13)) >> 25;
131 var2 = ((int64_t)s_calib.P8 * p) >> 19;
132 p = ((p + var1 + var2) >> 8) + ((int64_t)s_calib.P7 << 4);
133 return (float)p / 25600.0f; /* hPa */
134}

136static float compensate_humidity(int32_t raw_h)
137{
138 int32_t v = s_t_fine - 76800;
139 v = (((((raw_h << 14) - ((int32_t)s_calib.H4 << 20) -
140 ((int32_t)s_calib.H5 * v)) + 16384) >> 15) *
141 (((((((v * (int32_t)s_calib.H6) >> 10) *
142 (((v * (int32_t)s_calib.H3) >> 11) + 32768)) >> 10) +
143 2097152) * (int32_t)s_calib.H2 + 8192) >> 14));
144 v -= ((((v >> 15) * (v >> 15)) >> 7) * (int32_t)s_calib.H1) >> 4;
145 if (v < 0) v = 0;
146 if (v > 419430400) v = 419430400;
147 float rh = (float)(v >> 12) / 1024.0f;
148 if (rh < 0.0f) rh = 0.0f;
149 if (rh > 100.0f) rh = 100.0f;
150 return rh;
151}

155esp_err_t bme280_init(i2c_master_bus_handle_t bus, uint8_t i2c_addr)
156{
157 const i2c_device_config_t dev_cfg = {
158 .dev_addr_length = I2C_ADDR_BIT_LEN_7,
159 .device_address = i2c_addr,
160 .scl_speed_hz = 400000,
161 };
162 esp_err_t ret = i2c_master_bus_add_device(bus, &dev_cfg, &s_dev);
163 if (ret != ESP_OK) {
164 ESP_LOGE(TAG, "Ajout device I2C 0x%02X échoué : %s",
165 i2c_addr, esp_err_to_name(ret));
166 return ret;
167 }
168
169 /* Vérification chip ID */
170 uint8_t chip_id = 0;
171 ret = read_regs(REG_CHIP_ID, &chip_id, 1);
172 if (ret != ESP_OK || chip_id != BME280_CHIP_ID) {
173 ESP_LOGW(TAG, "BME280 non détecté (chip_id=0x%02X)", chip_id);
174 return (ret == ESP_OK) ? ESP_ERR_NOT_FOUND : ret;
175 }
176
177 /* Reset logiciel */
178 write_reg(REG_RESET, 0xB6);
179 vTaskDelay(pdMS_TO_TICKS(5));
180
181 /* Lecture calibration */
182 ret = read_calibration();
183 if (ret != ESP_OK) {
184 ESP_LOGE(TAG, "Lecture calibration échouée");
185 return ret;
186 }
187
188 /* Configuration :
189 * ctrl_hum : osrs_h = ×4 (101b = 5)
190 * ctrl_meas : osrs_t = ×2 (010b), osrs_p = ×4 (011b), mode = sleep (00b)
191 * config : filtre IIR = ×4 (010b), t_sb = 0.5 ms (000b)
192 */
193 write_reg(REG_CTRL_HUM, 0x05); /* osrs_h ×4 */
194 write_reg(REG_CONFIG, 0x10); /* filtre ×4 */
195 write_reg(REG_CTRL_MEAS, 0x54); /* osrs_t ×2, osrs_p ×4, sleep mode */
196
197 ESP_LOGI(TAG, "BME280 initialisé (@0x%02X, chip_id=0x60)", i2c_addr);
198 return ESP_OK;
199}

201esp_err_t bme280_read(float *temp_c, float *pressure_hpa, float *humidity_pct)
202{
203 if (!s_dev) return ESP_ERR_INVALID_STATE;
204
205 /* Déclencher une mesure forced (mode = 01b → écriture de ctrl_meas) */
206 uint8_t ctrl;
207 esp_err_t ret = read_regs(REG_CTRL_MEAS, &ctrl, 1);
208 if (ret != ESP_OK) return ret;
209
210 /* Set mode = forced (01) en conservant l'oversampling */
211 ctrl = (ctrl & 0xFC) | 0x01;
212 ret = write_reg(REG_CTRL_MEAS, ctrl);
213 if (ret != ESP_OK) return ret;
214
215 /* Attente fin de conversion */
216 vTaskDelay(pdMS_TO_TICKS(MEAS_DELAY_MS));
217
218 /* Lecture de 8 octets : P (3), T (3), H (2) */
219 uint8_t data[8] = {0};
220 ret = read_regs(REG_DATA_START, data, sizeof(data));
221 if (ret != ESP_OK) {
222 ESP_LOGW(TAG, "Lecture données échouée : %s", esp_err_to_name(ret));
223 return ret;
224 }
225
226 int32_t raw_p = ((int32_t)data[0] << 12) | ((int32_t)data[1] << 4) | (data[2] >> 4);
227 int32_t raw_t = ((int32_t)data[3] << 12) | ((int32_t)data[4] << 4) | (data[5] >> 4);
228 int32_t raw_h = ((int32_t)data[6] << 8) | data[7];
229
230 /* Compensation (ordre obligatoire : T en premier pour s_t_fine) */
231 *temp_c = compensate_temperature(raw_t);
232 *pressure_hpa = compensate_pressure(raw_p);
233 *humidity_pct = compensate_humidity(raw_h);
234
235 return ESP_OK;
236}
