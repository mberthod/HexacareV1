84uint8_t vl53l8cx_motion_indicator_set_resolution(
85 VL53L8CX_Configuration *p_dev,
86 VL53L8CX_Motion_Configuration *p_motion_config,
87 uint8_t resolution)
88{
89 uint8_t i, status = VL53L8CX_STATUS_OK;
90
91 switch(resolution)
92 {
93 case VL53L8CX_RESOLUTION_4X4:
94 for(i = 0; i < (uint8_t)VL53L8CX_RESOLUTION_4X4; i++)
95 {
96 p_motion_config->map_id[i] = (int8_t)i;
97 }
98 (void)memset(p_motion_config->map_id + 16, -1, 48);
99 break;
100
101 case VL53L8CX_RESOLUTION_8X8:
102 for(i = 0; i < (uint8_t)VL53L8CX_RESOLUTION_8X8; i++)
103 {
104 p_motion_config->map_id[i] = (int8_t)((((int8_t)
105 i % 8)/2) + (4*((int8_t)i/16)));
106 }
107 break;
108
109 default:
110 status |= VL53L8CX_STATUS_ERROR;
111 break;
112 }
113
114 if(status == VL53L8CX_STATUS_OK)
115 {
116 status |= vl53l8cx_dci_write_data(p_dev,
117 (uint8_t*)(p_motion_config),
118 VL53L8CX_DCI_MOTION_DETECTOR_CFG,
119 (uint16_t)sizeof(*p_motion_config));
