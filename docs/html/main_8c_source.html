60/* Contexte système — statique (BSS) */
61static sys_context_t s_ctx;
62
63void app_main(void)
64{
65 /* ================================================================
66 * 1. Infrastructure de base
67 * ================================================================ */
68 esp_err_t nvs_ret = nvs_flash_init();
69 if (nvs_ret == ESP_ERR_NVS_NO_FREE_PAGES ||
70 nvs_ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
71 ESP_LOGW(TAG, "NVS corrompu — effacement");
72 ESP_ERROR_CHECK(nvs_flash_erase());
73 nvs_ret = nvs_flash_init();
74 }
75 ESP_ERROR_CHECK(nvs_ret);
76 ESP_ERROR_CHECK(esp_event_loop_create_default());
77
78 /* ================================================================
79 * 2. Stockage SPIFFS + panic handler
80 * ================================================================ */
81 if (littlefs_manager_init() == ESP_OK) {
82 if (panic_handler_install() == ESP_OK)
83 panic_handler_check_on_boot();
84 } else {
85 ESP_LOGW(TAG, "SPIFFS indisponible — logs désactivés");
86 }
87
88 /* ================================================================
89 * 3. Diagnostic matériel (SPI LIDAR + PCA9555 LPn)
90 * ================================================================ */
91 memset(&s_ctx, 0, sizeof(s_ctx));
92 s_ctx.enviro.temp_hdc_c = NAN;
93 s_ctx.enviro.humidity_hdc = NAN;
94 s_ctx.enviro.temp_bme_c = NAN;
95 s_ctx.enviro.pressure_hpa = NAN;
96 s_ctx.enviro.humidity_bme = NAN;
97
98 hw_diag_result_t diag = hw_diag_run(&s_ctx);
99 if (diag & HW_DIAG_LIDAR_ALL) ESP_LOGE(TAG, "Aucun LIDAR — dégradé");
100 if (diag & HW_DIAG_LIDAR_PARTIAL) ESP_LOGW(TAG, "LIDARs partiels");
101 if (diag & HW_DIAG_RADAR_MISSING) ESP_LOGW(TAG, "Radar absent");
102
103 /* ================================================================
104 * 4. Alimentations PCA9555 — AVANT init des capteurs
105 * POWER_RADAR alimente HDC1080 + BME280 sur la carte radar.
106 * ================================================================ */
107#if LEXACARE_PWR_RADAR_ENABLED
108 if (hw_diag_pca9555_set_power(PCA9555_BIT_PWR_RADAR, true) == ESP_OK) {
109 ESP_LOGI(TAG, "POWER_RADAR (IO1.5) activé — capteurs I2C alimentés");
110 vTaskDelay(pdMS_TO_TICKS(50)); /* Stabilisation alimentation */
111 }
112#endif
113#if LEXACARE_PWR_FAN_ENABLED
114 hw_diag_pca9555_set_power(PCA9555_BIT_PWR_FAN, true);
115 ESP_LOGI(TAG, "POWER_FAN (IO1.4) activé");
116#endif
117#if LEXACARE_PWR_MLX_ENABLED
118 hw_diag_pca9555_set_power(PCA9555_BIT_PWR_MLX, true);
119 ESP_LOGI(TAG, "POWER_MLX (IO1.7) activé");
120 vTaskDelay(pdMS_TO_TICKS(20));
121#endif
122
123 /* ================================================================
124 * 5. Capteurs I2C environnementaux (carte radar, I2C_NUM_1)
125 * ================================================================ */
126 i2c_master_bus_handle_t sensors_bus = NULL;
127#if LEXACARE_ENABLE_HDC1080 || LEXACARE_ENABLE_BME280 || LEXACARE_ENABLE_MLX90640
128 if (hw_diag_init_sensor_bus(&sensors_bus) != ESP_OK) {
129 ESP_LOGW(TAG, "Bus I2C_NUM_1 indisponible — capteurs enviro off");
130 }
131#endif
132
133#if LEXACARE_ENABLE_HDC1080
134 if (sensors_bus) {
135 if (hdc1080_init(sensors_bus) == ESP_OK)
136 ESP_LOGI(TAG, "HDC1080 prêt @0x40");
137 else
138 ESP_LOGW(TAG, "HDC1080 non détecté @0x40");
139 }
140#endif
141
142#if LEXACARE_ENABLE_BME280
143 if (sensors_bus) {
144 if (bme280_init(sensors_bus, BME280_I2C_ADDR_DEFAULT) == ESP_OK) {
145 ESP_LOGI(TAG, "BME280 prêt @0x76");
146 } else {
147 ESP_LOGW(TAG, "BME280 @0x76 échoué — essai @0x77");
148 if (bme280_init(sensors_bus, BME280_I2C_ADDR_ALT) == ESP_OK)
149 ESP_LOGI(TAG, "BME280 prêt @0x77");
150 else
151 ESP_LOGW(TAG, "BME280 non détecté");
152 }
153 }
154#endif
155
156 /* ================================================================
157 * 6. Queues FreeRTOS + mutex
158 * ================================================================ */
159 s_ctx.data_mutex = xSemaphoreCreateMutex();
160 if (!s_ctx.data_mutex) { ESP_LOGE(TAG, "Mutex KO"); esp_restart(); }
161
162 s_ctx.sensor_to_ai_queue = xQueueCreate(4, sizeof(sensor_frame_t));
163 s_ctx.ai_to_mesh_queue = xQueueCreate(8, sizeof(ai_event_t));
164 s_ctx.diag_to_pc_queue = xQueueCreate(8, sizeof(ai_event_t));
165 s_ctx.log_queue = xQueueCreate(16, sizeof(char *));
166
167 if (!s_ctx.sensor_to_ai_queue || !s_ctx.ai_to_mesh_queue ||
168 !s_ctx.diag_to_pc_queue || !s_ctx.log_queue) {
169 ESP_LOGE(TAG, "Queues KO — redémarrage");
170 esp_restart();
171 }
172
173 /* ================================================================
174 * 7. Alimentation MIC — AVANT WiFi
175 * Toutes les opérations PCA9555 (allocation du device handle I2C)
176 * doivent être effectuées AVANT mesh_manager_init().
177 * Après WiFi, le heap peut être fragmenté → LoadStoreAlignment
178 * sur l'allocation dans i2c_master_bus_add_device.
179 * ================================================================ */
180#if LEXACARE_PWR_MIC_ENABLED && LEXACARE_ENABLE_MIC
181 if (hw_diag_pca9555_set_power(PCA9555_BIT_PWR_MIC, true) == ESP_OK) {
182 ESP_LOGI(TAG, "POWER_MIC (IO1.6) activé");
183 vTaskDelay(pdMS_TO_TICKS(10));
184 }
185#endif
186
187 /* ================================================================
188 * 8. Microphone I2S — AVANT WiFi (évite fragmentation heap TLSF)
189 * ================================================================ */
190#if LEXACARE_ENABLE_MIC
191 if (mic_driver_init() == ESP_OK)
192 ESP_LOGI(TAG, "Microphone I2S prêt");
193 else
194 ESP_LOGW(TAG, "Microphone I2S KO");
195#endif
196
197 /* ================================================================
198 * 9. Réseau mesh ESP-NOW — APRÈS toutes les opérations PCA9555
199 * ================================================================ */
200#if LEXACARE_ENABLE_MESH
201 ESP_ERROR_CHECK(mesh_manager_init(&s_ctx));
202#else
203 ESP_LOGW(TAG, "Mesh WiFi désactivé (LEXACARE_ENABLE_MESH=0)");
204#endif
205
206 /* ================================================================
207 * 10. Démarrage des tâches
208 * ================================================================ */
209#if LEXACARE_ENABLE_AI
210 ESP_ERROR_CHECK(ai_engine_task_start(&s_ctx)); /* Core 1, prio 9 */
211#else
212 ESP_LOGW(TAG, "IA désactivée (LEXACARE_ENABLE_AI=0)");
213#endif
214 ESP_ERROR_CHECK(lidar_driver_start(&s_ctx)); /* Core 1, prio 10 */
215#if LEXACARE_ENABLE_MESH
216 ESP_ERROR_CHECK(mesh_task_start(&s_ctx)); /* Core 0, prio 8 */
217#endif
218 ESP_ERROR_CHECK(pc_diag_task_start(&s_ctx)); /* Core 0, prio 3 */
219
220 /* ================================================================
221 * Rapport de démarrage
222 * ================================================================ */
223 ESP_LOGI(TAG, "================================================");
224 ESP_LOGI(TAG, " LexaCare V1 — Firmware opérationnel");
225 ESP_LOGI(TAG, " IDF : %s | Rôle : %s",
226 esp_get_idf_version(),
227 s_ctx.is_root_node ? "ROOT" : "NODE");
228 ESP_LOGI(TAG, " Heap interne : %lu B | PSRAM : %lu B",
229 (unsigned long)esp_get_free_heap_size(),
230 (unsigned long)heap_caps_get_free_size(MALLOC_CAP_SPIRAM));
231 ESP_LOGI(TAG, " Config: LIDAR×%d RADAR=%d HDC=%d BME=%d MIC=%d MESH=%d",
232 LEXACARE_LIDAR_COUNT,
233 LEXACARE_ENABLE_RADAR,
234 LEXACARE_ENABLE_HDC1080,
235 LEXACARE_ENABLE_BME280,
236 LEXACARE_ENABLE_MIC,
237 LEXACARE_ENABLE_MESH);
238 ESP_LOGI(TAG, "================================================");
239}
