15uint8_t vl53l8cx_get_detection_thresholds_enable(
16 VL53L8CX_Configuration *p_dev,
17 uint8_t *p_enabled)
18{
19 uint8_t status = VL53L8CX_STATUS_OK;
20
21 status |= vl53l8cx_dci_read_data(p_dev, (uint8_t*)p_dev->temp_buffer,
22 VL53L8CX_DCI_DET_THRESH_GLOBAL_CONFIG, 8);
23 *p_enabled = p_dev->temp_buffer[0x1];

28uint8_t vl53l8cx_set_detection_thresholds_enable(
29 VL53L8CX_Configuration *p_dev,
30 uint8_t enabled)
31{
32 uint8_t tmp, status = VL53L8CX_STATUS_OK;
33 uint8_t grp_global_config[] = {0x01, 0x00, 0x01, 0x00};
34
35 if(enabled == (uint8_t)1)
36 {
37 grp_global_config[0x01] = 0x01;
38 tmp = 0x04;
39 }
40 else
41 {
42 grp_global_config[0x01] = 0x00;
43 tmp = 0x0C;
44 }
45
46 /* Set global interrupt config */
47 status |= vl53l8cx_dci_replace_data(p_dev, p_dev->temp_buffer,
48 VL53L8CX_DCI_DET_THRESH_GLOBAL_CONFIG, 8,
49 (uint8_t*)&grp_global_config, 4, 0x00);
50
51 /* Update interrupt config */
52 status |= vl53l8cx_dci_replace_data(p_dev, p_dev->temp_buffer,
53 VL53L8CX_DCI_DET_THRESH_CONFIG, 20,
54 (uint8_t*)&tmp, 1, 0x11);
55
56 return status;
57}

59uint8_t vl53l8cx_get_detection_thresholds(
60 VL53L8CX_Configuration *p_dev,
61 VL53L8CX_DetectionThresholds *p_thresholds)
62{
63 uint8_t i, status = VL53L8CX_STATUS_OK;
64
65 /* Get thresholds configuration */
66 status |= vl53l8cx_dci_read_data(p_dev, (uint8_t*)p_thresholds,
67 VL53L8CX_DCI_DET_THRESH_START,
68 (uint16_t)VL53L8CX_NB_THRESHOLDS
69 *(uint16_t)sizeof(VL53L8CX_DetectionThresholds));
70
71 for(i = 0; i < (uint8_t)VL53L8CX_NB_THRESHOLDS; i++)
72 {
73 switch(p_thresholds[i].measurement)
74 {
75 case VL53L8CX_DISTANCE_MM:
76 p_thresholds[i].param_low_thresh /= 4;
77 p_thresholds[i].param_high_thresh /= 4;
78 break;
79 case VL53L8CX_SIGNAL_PER_SPAD_KCPS:
80 p_thresholds[i].param_low_thresh /= 2048;
81 p_thresholds[i].param_high_thresh /= 2048;
82 break;
83 case VL53L8CX_RANGE_SIGMA_MM:
84 p_thresholds[i].param_low_thresh /= 128;
85 p_thresholds[i].param_high_thresh /= 128;
86 break;
87 case VL53L8CX_AMBIENT_PER_SPAD_KCPS:
88 p_thresholds[i].param_low_thresh /= 2048;
89 p_thresholds[i].param_high_thresh /= 2048;
90 break;
91 case VL53L8CX_NB_SPADS_ENABLED:
92 p_thresholds[i].param_low_thresh /= 256;
93 p_thresholds[i].param_high_thresh /= 256;
94 break;
95 case VL53L8CX_MOTION_INDICATOR:
96 p_thresholds[i].param_low_thresh /= 65535;
97 p_thresholds[i].param_high_thresh /= 65535;
98 break;
99 default:
100 break;
101 }
102 }
103
104 return status;
105}

107uint8_t vl53l8cx_set_detection_thresholds(
108 VL53L8CX_Configuration *p_dev,
109 VL53L8CX_DetectionThresholds *p_thresholds)
110{
111 uint8_t i, status = VL53L8CX_STATUS_OK;
112 uint8_t grp_valid_target_cfg[] = {0x05, 0x05, 0x05, 0x05,
113 0x05, 0x05, 0x05, 0x05};
114
115 for(i = 0; i < (uint8_t) VL53L8CX_NB_THRESHOLDS; i++)
116 {
117 switch(p_thresholds->measurement)
118 {
119 case VL53L8CX_DISTANCE_MM:
120 p_thresholds[i].param_low_thresh *= 4;
121 p_thresholds[i].param_high_thresh *= 4;
122 break;
123 case VL53L8CX_SIGNAL_PER_SPAD_KCPS:
124 p_thresholds[i].param_low_thresh *= 2048;
125 p_thresholds[i].param_high_thresh *= 2048;
126 break;
127 case VL53L8CX_RANGE_SIGMA_MM:
128 p_thresholds[i].param_low_thresh *= 128;
129 p_thresholds[i].param_high_thresh *= 128;
130 break;
131 case VL53L8CX_AMBIENT_PER_SPAD_KCPS:
132 p_thresholds[i].param_low_thresh *= 2048;
133 p_thresholds[i].param_high_thresh *= 2048;
134 break;
135 case VL53L8CX_NB_SPADS_ENABLED:
136 p_thresholds[i].param_low_thresh *= 256;
137 p_thresholds[i].param_high_thresh *= 256;
138 break;
139 case VL53L8CX_MOTION_INDICATOR:
140 p_thresholds[i].param_low_thresh *= 65535;
141 p_thresholds[i].param_high_thresh *= 65535;
142 break;
143 default:
144 break;
145 }
146 }
147
148 /* Set valid target list */
149 status |= vl53l8cx_dci_write_data(p_dev, (uint8_t*)grp_valid_target_cfg,
150 VL53L8CX_DCI_DET_THRESH_VALID_STATUS,
151 (uint16_t)sizeof(grp_valid_target_cfg));
152
153 /* Set thresholds configuration */
154 status |= vl53l8cx_dci_write_data(p_dev, (uint8_t*)p_thresholds,
155 VL53L8CX_DCI_DET_THRESH_START,
156 (uint16_t)(VL53L8CX_NB_THRESHOLDS
157 *sizeof(VL53L8CX_DetectionThresholds)));
158
159 return status;
160}
