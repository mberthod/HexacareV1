4 * @brief Couche plateforme VL53L8CX — implémentation SPI via ESP-IDF.
5 *
6 * Protocole SPI VL53L8CX (mode 3, CPOL=1 CPHA=1) :
7 * Écriture N octets : TX=[addr_hi & 0x7F][addr_lo][d0..dN-1]
8 * Lecture N octets : TX=[addr_hi | 0x80][addr_lo][0x00 × N]
9 * RX=[xx][xx][d0..dN-1] (2 premiers octets ignorés)
10 *
11 * Les handles SPI sont créés dans hw_diag.c (spi_bus_add_device) et
12 * transmis via sys_context_t → VL53L8CX_Platform.spi_dev.
13 */

40/* ================================================================
41 * spi_xfer (interne)
42 * @brief Exécute un transfert SPI full-duplex avec allocation dynamique.
43 *
44 * @param spi Handle SPI ESP-IDF du capteur.
45 * @param tx_buf Buffer TX (header + données, ou header + dummy pour lecture).
46 * @param rx_buf Buffer RX (même taille que TX).
47 * @param length Nombre total d'octets à transférer.
48 * @return ESP_OK si succès.
49 * ================================================================ */
50static esp_err_t spi_xfer(spi_device_handle_t spi,
51 const uint8_t *tx_buf, uint8_t *rx_buf,
52 size_t length)
53{
54 spi_transaction_t t = {
55 .length = length * 8, /* en bits */
56 .tx_buffer = tx_buf,
57 .rx_buffer = rx_buf,
58 };
59 return spi_device_polling_transmit(spi, &t);
60}

62/* ================================================================
63 * VL53L8CX_RdMulti
64 * ================================================================ */
65uint8_t VL53L8CX_RdMulti(VL53L8CX_Platform *p_platform,
66 uint16_t RegisterAddress,
67 uint8_t *p_values, uint32_t size)
68{
69 size_t total = 2 + (size_t)size;
70 uint8_t *tx = calloc(1, total);
71 uint8_t *rx = calloc(1, total);
72
73 if (!tx || !rx) {
74 ESP_LOGE(TAG, "RdMulti : malloc(%u) échoué", (unsigned)total);
75 free(tx); free(rx);
76 return 1;
77 }
78
79 /* Header de lecture : bit 15 = 1 */
80 tx[0] = (uint8_t)((RegisterAddress >> 8) | 0x80);
81 tx[1] = (uint8_t)(RegisterAddress & 0xFF);
82 /* Octets restants laissés à 0x00 (dummy pour générer l'horloge) */
83
84 esp_err_t ret = spi_xfer(p_platform->spi_dev, tx, rx, total);
85
86 if (ret == ESP_OK) {
87 memcpy(p_values, &rx[2], size); /* Sauter les 2 octets header */
88 } else {
89 ESP_LOGE(TAG, "RdMulti addr=0x%04X : %s",
90 RegisterAddress, esp_err_to_name(ret));
91 }
92
93 free(tx);
94 free(rx);
95 return (ret == ESP_OK) ? 0 : 1;
96}

98/* ================================================================
99 * VL53L8CX_WrMulti
100 * ================================================================ */
101uint8_t VL53L8CX_WrMulti(VL53L8CX_Platform *p_platform,
102 uint16_t RegisterAddress,
103 uint8_t *p_values, uint32_t size)
104{
105 size_t total = 2 + (size_t)size;
106 uint8_t *tx = malloc(total);
107
108 if (!tx) {
109 ESP_LOGE(TAG, "WrMulti : malloc(%u) échoué", (unsigned)total);
110 return 1;
111 }
112
113 /* Header d'écriture : bit 15 = 0 */
114 tx[0] = (uint8_t)((RegisterAddress >> 8) & 0x7F);
115 tx[1] = (uint8_t)(RegisterAddress & 0xFF);
116 memcpy(&tx[2], p_values, size);
117
118 esp_err_t ret = spi_xfer(p_platform->spi_dev, tx, NULL, total);
119
120 if (ret != ESP_OK) {
121 ESP_LOGE(TAG, "WrMulti addr=0x%04X : %s",
122 RegisterAddress, esp_err_to_name(ret));
123 }
124
125 free(tx);
126 return (ret == ESP_OK) ? 0 : 1;
127}
