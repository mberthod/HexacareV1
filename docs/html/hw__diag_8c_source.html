119/* ================================================================
120 * diag_init_spi_lidars (interne)
121 * @brief Initialise le bus SPI et ajoute les 4 devices LIDAR.
122 *
123 * Chaque LIDAR a son propre NCS GPIO → handle SPI individuel stocké
124 * dans ctx->lidar_spi[i] pour réutilisation par sensor_acq.
125 *
126 * @param ctx Contexte système (lidar_spi[] rempli en sortie).
127 * @return ESP_OK si succès.
128 * ================================================================ */
129static esp_err_t diag_init_spi_lidars(sys_context_t *ctx)
130{
131 const spi_bus_config_t bus_cfg = {
132 .mosi_io_num = PIN_LIDAR_MOSI,
133 .miso_io_num = PIN_LIDAR_MISO,
134 .sclk_io_num = PIN_LIDAR_CLK,
135 .quadwp_io_num = -1,
136 .quadhd_io_num = -1,
137 .max_transfer_sz = LIDAR_SPI_MAX_XFER,
138 };
139
140 esp_err_t ret = spi_bus_initialize(LIDAR_SPI_HOST, &bus_cfg, SPI_DMA_CH_AUTO);
141 if (ret != ESP_OK) {
142 ESP_LOGE(TAG, "SPI bus init échoué : %s", esp_err_to_name(ret));
143 return ret;
144 }
145
146 ESP_LOGI(TAG, "Bus SPI LIDAR initialisé (CLK=%d, MOSI=%d, MISO=%d, %d MHz)",
147 PIN_LIDAR_CLK, PIN_LIDAR_MOSI, PIN_LIDAR_MISO,
148 LIDAR_SPI_FREQ_HZ / 1000000);
149
150 for (int i = 0; i < LIDAR_NUM_FRONT; i++) {
151 const spi_device_interface_config_t dev_cfg = {
152 .mode = 3, /* CPOL=1, CPHA=1 (VL53L8CX SPI) */
153 .clock_speed_hz = LIDAR_SPI_FREQ_HZ,
154 .spics_io_num = s_ncs_pins[i], /* NCS dédié par capteur */
155 .queue_size = 1,
156 };
157
158 ret = spi_bus_add_device(LIDAR_SPI_HOST, &dev_cfg, &ctx->lidar_spi[i]);
159 if (ret != ESP_OK) {
160 ESP_LOGE(TAG, "SPI add device LIDAR[%d] (NCS=GPIO%d) échoué : %s",
161 i, s_ncs_pins[i], esp_err_to_name(ret));
162 return ret;
163 }
164
165 ESP_LOGI(TAG, "LIDAR[%d] SPI device ajouté (NCS=GPIO%d)",
166 i, s_ncs_pins[i]);
167 }
168
169 return ESP_OK;
170}

172/* ================================================================
173 * diag_spi_ping_lidar (interne)
174 * @brief Vérifie la présence d'un VL53L8CX via lecture du Model ID.
175 *
176 * Lit le registre 0x7FFF qui doit retourner 0xF0 si le capteur répond.
177 * En STUB mode (sans ULD), retourne true si la transaction SPI réussit
178 * et que MISO n'est pas en flottant (0xFF = absent probable).
179 *
180 * @param spi Handle SPI du capteur.
181 * @return true si capteur détecté.
182 * ================================================================ */
183static bool diag_spi_ping_lidar(spi_device_handle_t spi)
184{
185 /* Lecture registre 0x7FFF : TX=[0xFF][0xFF][dummy], RX=[xx][xx][model_id]
186 * (bit15=1 → lecture, adresse 0x7FFF → addr_hi=0x7F|0x80=0xFF, addr_lo=0xFF) */
187 const uint8_t tx[3] = {0xFF, 0xFF, 0x00};
188 uint8_t rx[3] = {0x00, 0x00, 0x00};
189
190 spi_transaction_t t = {
191 .length = sizeof(tx) * 8,
192 .tx_buffer = tx,
193 .rx_buffer = rx,
194 };
195
196 esp_err_t ret = spi_device_polling_transmit(spi, &t);
197 if (ret != ESP_OK) {
198 return false;
199 }
200
201 /* 0xFF = MISO flottant (absent) ; 0xF0 = Model ID VL53L8CX */
202 return (rx[2] != 0xFF);
203}

205/* ================================================================
206 * pca9555_write_reg (interne)
207 * @brief Écrit un octet dans un registre du PCA9555D via I2C.
208 *
209 * Utilise s_pca_dev créé UNE SEULE FOIS par pca9555_init().
210 * Pas d'allocation heap à chaque appel → safe après WiFi init.
211 *
212 * @param bus Ignoré (conservé pour compat. — utilise s_pca_dev).
213 * @param reg Registre cible.
214 * @param val Valeur à écrire.
215 * @return ESP_OK si succès.
216 * ================================================================ */
217static esp_err_t pca9555_write_reg(i2c_master_bus_handle_t bus,
218 uint8_t reg, uint8_t val)
219{
220 (void)bus; /* s_pca_dev est utilisé directement */
221
222 if (!s_pca_dev) {
223 ESP_LOGE(TAG, "pca9555_write_reg : device non initialisé");
224 return ESP_ERR_INVALID_STATE;
225 }
226
227 const uint8_t tx[2] = {reg, val};
228 return i2c_master_transmit(s_pca_dev, tx, sizeof(tx),
229 pdMS_TO_TICKS(I2C_TIMEOUT_MS));
230}

232/* ================================================================
233 * pca9555_init (interne)
234 * @brief Initialise le PCA9555D : Port 0 + Port 1 en sortie, tout à 0.
235 *
236 * Port 0 : LPn LIDAR 0–4 (tous à 0 = reset global).
237 * Port 1 : alimentation sous-systèmes (tous à 0 = éteints).
238 *
239 * @param bus Handle du bus I2C maître.
240 * @return ESP_OK si succès.
241 * ================================================================ */
242static esp_err_t pca9555_init(i2c_master_bus_handle_t bus)
243{
244 /* Création du device handle PCA9555 — UNE SEULE FOIS.
245 * Alloué ici (avant WiFi) pour éviter tout problème de heap après.
246 * Réutilisé par pca9555_write_reg() sans nouvelle allocation. */
247 if (!s_pca_dev) {
248 const i2c_device_config_t dev_cfg = {
249 .dev_addr_length = I2C_ADDR_BIT_LEN_7,
250 .device_address = PCA9555_I2C_ADDR,
251 .scl_speed_hz = I2C_BUS_FREQ_HZ,
252 };
253 esp_err_t ret_dev = i2c_master_bus_add_device(bus, &dev_cfg, &s_pca_dev);
254 if (ret_dev != ESP_OK) {
255 ESP_LOGE(TAG, "PCA9555 add_device échoué : %s", esp_err_to_name(ret_dev));
256 return ret_dev;
257 }
258 }
259
260 /* --- Port 0 ---
261 * IO0.0 = sortie (DIST_SHUTDOWN)
262 * IO0.1, IO0.2 = entrée Hi-Z (GPIO radar LD6002, non utilisés)
263 * IO0.3 – IO0.7 = sorties (LPn_4, LPn_3, LPn_2, LPn_1, RADAR_LPN)
264 * Config : 0 = sortie → PCA9555_CFG0_MASK = 0x06 (bits 1 et 2 en entrée) */
265 esp_err_t ret = pca9555_write_reg(bus, PCA9555_REG_CONFIG_0, PCA9555_CFG0_MASK);
266 if (ret != ESP_OK) {
267 ESP_LOGE(TAG, "PCA9555 config Port0 échoué : %s", esp_err_to_name(ret));
268 return ret;
269 }
270
271 /* Valeur initiale Port 0 :
272 * IO0.0 (DIST_SHUTDOWN) = 0 → VL53L0X actif (non mis en veille)
273 * IO0.3–IO0.7 (LPn + RADAR_LPN) = 0 → tous les LIDARs en reset */
274 s_pca_out = 0x00;
275 ret = pca9555_write_reg(bus, PCA9555_REG_OUTPUT_0, s_pca_out);
276 if (ret != ESP_OK) {
277 ESP_LOGE(TAG, "PCA9555 output Port0 échoué : %s", esp_err_to_name(ret));
278 return ret;
279 }
280
281 /* --- Port 1 ---
282 * IO1.0–IO1.3 = entrée Hi-Z (non utilisés)
283 * IO1.4–IO1.7 = sorties (POWER_FAN, POWER_RADAR, POWER_MIC, POWER_MLX)
284 * Config : PCA9555_CFG1_MASK = 0x0F (bits 0–3 en entrée, bits 4–7 en sortie) */
285 ret = pca9555_write_reg(bus, PCA9555_REG_CONFIG_1, PCA9555_CFG1_MASK);
286 if (ret != ESP_OK) {
287 ESP_LOGE(TAG, "PCA9555 config Port1 échoué : %s", esp_err_to_name(ret));
288 return ret;
289 }
290
291 /* Valeur initiale Port 1 : toutes alimentations éteintes */
292 ret = pca9555_write_reg(bus, PCA9555_REG_OUTPUT_1, 0x00);
293 if (ret != ESP_OK) {
294 ESP_LOGE(TAG, "PCA9555 output Port1 échoué : %s", esp_err_to_name(ret));
295 return ret;
296 }
297
298 ESP_LOGI(TAG, "PCA9555 initialisé (@0x%02X, SDA=%d, SCL=%d) — Port0/1 configurés",
299 PCA9555_I2C_ADDR, PIN_I2C0_SDA, PIN_I2C0_SCL);
300 return ESP_OK;
301}

303/* ================================================================
304 * pca9555_set_lpn (interne)
305 * @brief Active ou désactive le LPn d'un capteur via le PCA9555.
306 *
307 * @param bus Handle du bus I2C_NUM_0.
308 * @param idx Index du capteur (1–4).
309 * @param active true → LPn=1 (actif), false → LPn=0 (reset).
310 * @return ESP_OK si succès.
311 * ================================================================ */
312static esp_err_t pca9555_set_lpn(i2c_master_bus_handle_t bus,
313 int idx, bool active)
314{
315 /* idx est un index tableau (0 = LIDAR 1 hardware, 3 = LIDAR 4 hardware) */
316 if (idx < 0 || idx >= LIDAR_NUM_FRONT) {
317 return ESP_ERR_INVALID_ARG;
318 }
319
320 if (active) {
321 s_pca_out |= s_pca_lpn_bit[idx];
322 } else {
323 s_pca_out &= ~s_pca_lpn_bit[idx];
324 }
325 return pca9555_write_reg(bus, PCA9555_REG_OUTPUT_0, s_pca_out);
326}

328/* ================================================================
329 * diag_scan_lidars (interne)
330 * @brief Active les LIDARs un par un via LPn (PCA9555) et les teste en SPI.
331 *
332 * Tous les LPn sont gérés par le PCA9555 :
333 * LIDAR 0 : IO7, carte radar — non connecté, ignoré.
334 * LIDAR 1–3 : IO3/IO5/IO4.
335 * Pas de réassignation d'adresse : chaque capteur a son NCS SPI dédié.
336 *
337 * @param ctx Contexte système (lidar_ok[] rempli en sortie).
338 * @param pca_bus Handle du bus I2C PCA9555 (NULL = PCA9555 non disponible).
339 * ================================================================ */
340static void diag_scan_lidars(sys_context_t *ctx,
341 i2c_master_bus_handle_t pca_bus)
342{
343 /* Initialisation PCA9555 : Port 0 + Port 1 en sortie, tout à 0 */
344 if (pca_bus) {
345 esp_err_t ret = pca9555_init(pca_bus);
346 if (ret != ESP_OK) {
347 ESP_LOGE(TAG, "PCA9555 inaccessible — LPn non contrôlables");
348 pca_bus = NULL;
349 }
350 }
351
352 /* --- Validation bus SPI AVANT toute manipulation LPN ─────────────────
353 * Envoie une trame de test sur NCS0 (LIDAR 1) pour confirmer que
354 * GPIO17 (CLK) est bien piloté par le contrôleur SPI.
355 * Le VL53L8CX ne répondra pas (LPN=0, pas de firmware ULD chargé)
356 * mais le CLK doit impérativement bouger. ----------------------------*/
357 {
358 const uint8_t tx_test[3] = {0xFF, 0xFF, 0x00};
359 uint8_t rx_test[3] = {0};
360 spi_transaction_t t_test = {
361 .length = sizeof(tx_test) * 8,
362 .tx_buffer = tx_test,
363 .rx_buffer = rx_test,
364 };
365 esp_err_t spi_test = spi_device_polling_transmit(ctx->lidar_spi[0], &t_test);
366 if (spi_test == ESP_OK) {
367 ESP_LOGI(TAG, "Bus SPI LIDAR validé — GPIO%d (CLK) fonctionne",
368 PIN_LIDAR_CLK);
369 } else {
370 ESP_LOGE(TAG, "Bus SPI LIDAR ERREUR : %s — vérifier GPIO%d",
371 esp_err_to_name(spi_test), PIN_LIDAR_CLK);
372 }
373 }
374
375 vTaskDelay(pdMS_TO_TICKS(10)); /* Stabilisation après reset global */
376
377 /* --- Scan séquentiel des 4 LIDARs (index 0–3 = LIDAR 1–4 hardware) ---
378 *
379 * RÈGLE : le ping SPI est TOUJOURS exécuté, même si LPN échoue.
380 * La CLK (GPIO17) doit bouger à chaque itération quelle que soit
381 * l'état du PCA9555. Le LIDAR ne répondra que si LPN=1 ET ULD chargé.
382 * -------------------------------------------------------------------*/
383 for (int i = 0; i < LIDAR_NUM_FRONT; i++) {
384
385 /* Activation LPn via PCA9555 — non bloquante pour le SPI */
386 bool lpn_ok = false;
387 if (pca_bus) {
388 esp_err_t lpn_ret = pca9555_set_lpn(pca_bus, i, true);
389 if (lpn_ret == ESP_OK) {
390 lpn_ok = true;
391 } else {
392 ESP_LOGW(TAG, "LIDAR%d : PCA9555 LPn échoué (%s) — SPI testé quand même",
393 i + 1, esp_err_to_name(lpn_ret));
394 }
395 }
396
397 vTaskDelay(pdMS_TO_TICKS(LIDAR_LPN_DELAY_MS));
398
399 /* Test SPI inconditionnnel — génère toujours des impulsions sur CLK */
400 ctx->lidar_ok[i] = diag_spi_ping_lidar(ctx->lidar_spi[i]);
401
402 ESP_LOGI(TAG, "LIDAR%d SPI %-6s | NCS=GPIO%-2d | LPn=IO0.%d (%s)",
403 i + 1,
404 ctx->lidar_ok[i] ? "OK" : "absent",
405 s_ncs_pins[i],
406 6 - i, /* LPn1=IO0.6, LPn2=IO0.5, LPn3=IO0.4, LPn4=IO0.3 */
407 lpn_ok ? "actif" : "échec");
408 }
409}

411/* ================================================================
412 * diag_test_radar (interne)
413 * @brief Configure UART2 et attend une trame TinyFrame valide (SOF=0x01).
414 *
415 * @param ctx Contexte système (radar_ok rempli).
416 * ================================================================ */
417static void diag_test_radar(sys_context_t *ctx)
418{
419 const uart_config_t uart_cfg = {
420 .baud_rate = RADAR_BAUD_RATE,
421 .data_bits = UART_DATA_8_BITS,
422 .parity = UART_PARITY_DISABLE,
423 .stop_bits = UART_STOP_BITS_1,
424 .flow_ctrl = UART_HW_FLOWCTRL_DISABLE,
425 .source_clk = UART_SCLK_DEFAULT,
426 };
427
428 ESP_ERROR_CHECK(uart_driver_install(RADAR_UART_PORT,
429 RADAR_RX_BUF_SIZE, 0, 0, NULL, 0));
430 ESP_ERROR_CHECK(uart_param_config(RADAR_UART_PORT, &uart_cfg));
431 ESP_ERROR_CHECK(uart_set_pin(RADAR_UART_PORT,
432 PIN_RADAR_TX, PIN_RADAR_RX,
433 UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE));
434
435 ESP_LOGI(TAG, "Attente trame LD6002 (%d ms, %d bauds)...",
436 RADAR_DETECT_TIMEOUT_MS, RADAR_BAUD_RATE);
437
438 uint8_t buf[RADAR_RX_BUF_SIZE];
439 TickType_t deadline = xTaskGetTickCount() +
440 pdMS_TO_TICKS(RADAR_DETECT_TIMEOUT_MS);
441 ctx->radar_ok = false;
442
443 while (xTaskGetTickCount() < deadline) {
444 int len = uart_read_bytes(RADAR_UART_PORT, buf, sizeof(buf),
445 pdMS_TO_TICKS(100));
446 if (len <= 0) {
447 continue;
448 }
449 /* Recherche du SOF TinyFrame (0x01) dans les données reçues */
450 for (int i = 0; i < len; i++) {
451 if (buf[i] == TF_SOF && (len - i) >= 5) {
452 ctx->radar_ok = true;
453 ESP_LOGI(TAG, "Radar LD6002 : trame TinyFrame détectée (%d octets reçus)",
454 len);
455 goto radar_done;
456 }
457 }
458 }
459
460radar_done:
461 if (!ctx->radar_ok) {
462 ESP_LOGW(TAG, "Radar LD6002 : aucune trame reçue dans le délai imparti");
463 }
464
465 /* Le driver UART reste installé (réutilisé par radar_driver dans sensor_acq) */
466}

501/* ================================================================
502 * hw_diag_run
503 * @brief Point d'entrée du diagnostic matériel.
504 *
505 * @param ctx Pointeur vers le contexte système à remplir.
506 * @return Masque hw_diag_result_t (0 = tout OK).
507 * ================================================================ */
508hw_diag_result_t hw_diag_run(sys_context_t *ctx)
509{
510 ESP_LOGI(TAG, "=== Démarrage du diagnostic matériel ===");
511
512 /* --- Bus SPI LIDAR : initialise le bus + 4 devices NCS dédiés --- */
513 ESP_ERROR_CHECK(diag_init_spi_lidars(ctx));
514
515 /* --- Bus I2C PCA9555 : PERSISTANT (contrôle LPn + alimentation) ---
516 * Ne pas détruire ce bus — il est réutilisé par hw_diag_pca9555_set_power()
517 * pour piloter les alimentations (POWER_MIC, POWER_RADAR, etc.) */
518 esp_err_t pca_ret = diag_init_i2c_bus(I2C_NUM_0,
519 PIN_I2C0_SDA, PIN_I2C0_SCL,
520 &s_pca_bus);
521 if (pca_ret != ESP_OK) {
522 ESP_LOGW(TAG, "PCA9555 I2C init échoué — LPn et alimentations non contrôlables");
523 }
524 i2c_master_bus_handle_t pca_bus = s_pca_bus; /* alias local pour lisibilité */
525
526 /* --- Lecture du sélecteur ROOT/NODE --- */
527 gpio_set_pull_mode(PIN_ROOT_SEL, GPIO_PULLUP_ONLY);
528 gpio_set_direction(PIN_ROOT_SEL, GPIO_MODE_INPUT);
529 ctx->is_root_node = (gpio_get_level(PIN_ROOT_SEL) == 0);
530 ESP_LOGI(TAG, "Rôle nœud : %s", ctx->is_root_node ? "ROOT (passerelle)" : "NODE");
531
532 /* --- Scan des LIDARs via SPI --- */
533 memset(ctx->lidar_ok, 0, sizeof(ctx->lidar_ok));
534 diag_scan_lidars(ctx, pca_bus);
535
536 /* Bus PCA9555 conservé dans s_pca_bus — réutilisé par hw_diag_pca9555_set_power() */
537
538 /* --- Test du radar (uniquement si connecté) --- */
539#if LEXACARE_ENABLE_RADAR
540 diag_test_radar(ctx);
541#else
542 ctx->radar_ok = false;
543 ESP_LOGI(TAG, "Radar LD6002 ignoré (LEXACARE_ENABLE_RADAR=0) — boot +2 s économisés");
544#endif
545
546 /* --- Calcul du masque de résultat --- */
547 hw_diag_result_t result = HW_DIAG_OK;
548 int lidar_count = 0;
549 for (int i = 0; i < LIDAR_NUM_FRONT; i++) {
550 if (ctx->lidar_ok[i]) lidar_count++;
551 }
552
553 if (lidar_count == 0) {
554 result |= HW_DIAG_LIDAR_ALL;
555 } else if (lidar_count < LIDAR_NUM_FRONT) {
556 result |= HW_DIAG_LIDAR_PARTIAL;
557 }
558
559#if LEXACARE_ENABLE_RADAR
560 if (!ctx->radar_ok) {
561 result |= HW_DIAG_RADAR_MISSING;
562 }
563#endif
564
565 /* Rapport JSON sur la console */
566 diag_print_json_report(ctx, result);
567
568 ESP_LOGI(TAG, "=== Diagnostic terminé (mask=0x%02X) ===", result);
569 return result;
570}
