1/**
2 * @file lidar_driver.c
3 * @ingroup group_sensor_acq
4 * @brief Acquisition LIDAR VL53L8CX — fusion 8×32 sur double bus I2C.
5 *
6 * Architecture dual-bus (contrainte débit) :
7 * I2C_NUM_0 → capteurs 0 & 1 (lus en parallèle avec bus 1)
8 * I2C_NUM_1 → capteurs 2, 3 & 4
9 *
10 * Fusion : capteur N → colonnes [N*8 .. N*8+7] de lidar_matrix_t.
11 * Cadence : 15 Hz via vTaskDelayUntil (période 66 ms).
12 * TWDT : esp_task_wdt_reset() à chaque itération.
13 *
14 * Dépendance ULD :
15 * Les appels vl53l8cx_*() nécessitent l'ULD VL53L8CX de ST.
16 * Voir instructions dans vl53l8cx_platform.h.
17 * Sans l'ULD, compiler avec -DLIDAR_STUB_MODE pour utiliser
18 * des données simulées (développement sans matériel).
19 */

99/* ================================================================
100 * lidar_init_sensor (interne)
101 * @brief Initialise un capteur VL53L8CX via SPI.
102 *
103 * Configure la résolution 8×8 et démarre le ranging continu.
104 * Le handle SPI est déjà créé par hw_diag_run() (spi_bus_add_device).
105 *
106 * @param dev Contexte du capteur.
107 * @param spi_dev Handle SPI ESP-IDF du capteur (NCS dédié).
108 * @param idx Index du capteur (0–3).
109 * @return true si l'initialisation réussit.
110 * ================================================================ */
111static bool lidar_init_sensor(lidar_dev_t *dev,
112 spi_device_handle_t spi_dev,
113 int idx)
114{
115 dev->platform.spi_dev = spi_dev;
116
117 if (VL53L8CX_PlatformInit(&dev->platform) != 0) {
118 ESP_LOGE(TAG, "LIDAR[%d] : échec PlatformInit SPI", idx);
119 return false;
120 }
121
122#if !LEXA_LIDAR_STUB
123 memset(&dev->uld, 0, sizeof(dev->uld));
124 dev->uld.platform = dev->platform;
125
126 if (vl53l8cx_init(&dev->uld) != VL53L8CX_STATUS_OK) {
127 ESP_LOGE(TAG, "LIDAR[%d] : vl53l8cx_init échoué", idx);
128 return false;
129 }
130
131 if (vl53l8cx_set_resolution(&dev->uld, VL53L8CX_RESOLUTION_8X8) != VL53L8CX_STATUS_OK) {
132 ESP_LOGE(TAG, "LIDAR[%d] : set_resolution échoué", idx);
133 return false;
134 }
135
136 if (vl53l8cx_set_ranging_frequency_hz(&dev->uld, LIDAR_FREQ_HZ) != VL53L8CX_STATUS_OK) {
137 ESP_LOGE(TAG, "LIDAR[%d] : set_ranging_frequency(%d Hz) échoué", idx, LIDAR_FREQ_HZ);
138 return false;
139 }
140
141 if (vl53l8cx_set_ranging_mode(&dev->uld, VL53L8CX_RANGING_MODE_AUTONOMOUS) != VL53L8CX_STATUS_OK) {
142 ESP_LOGE(TAG, "LIDAR[%d] : set_ranging_mode échoué", idx);
143 return false;
144 }
145
146 if (vl53l8cx_start_ranging(&dev->uld) != VL53L8CX_STATUS_OK) {
147 ESP_LOGE(TAG, "LIDAR[%d] : start_ranging échoué", idx);
148 return false;
149 }
150
151 ESP_LOGI(TAG, "LIDAR[%d] SPI : ULD ST actif (8×8, %d Hz)", idx, LIDAR_FREQ_HZ);
152#else
153 ESP_LOGW(TAG, "LIDAR[%d] SPI : STUB actif (LEXACARE_LIDAR_USE_ST_ULD=0)", idx);
154#endif
155
156 dev->active = true;
157 return true;
158}

160/* ================================================================
161 * lidar_read_sensor (interne)
162 * @brief Lit les données de ranging d'un capteur et les fusionne.
163 *
164 * Sans ULD : génère des données simulées pour valider l'architecture.
165 *
166 * @param dev Contexte du capteur.
167 * @param matrix Matrice destination.
168 * @param col_offset Décalage colonne (sensor_idx * 8).
169 * ================================================================ */
170static void lidar_read_sensor(lidar_dev_t *dev,
171 lidar_matrix_t *matrix,
172 int col_offset)
173{
174 if (!dev->active) return;
175
176#if !LEXA_LIDAR_STUB
177 uint8_t ready = 0;
178 if (vl53l8cx_check_data_ready(&dev->uld, &ready) != VL53L8CX_STATUS_OK || !ready) {
179 return;
180 }
181
182 if (vl53l8cx_get_ranging_data(&dev->uld, &dev->results) != VL53L8CX_STATUS_OK) {
183 return;
184 }
185
186 for (int zone = 0; zone < 64; zone++) {
187 int row = zone / 8;
188 int col = zone % 8;
189 /* ULD ST : distance_mm est un tableau 1D de int16_t.
190 * Si NB_TARGET_PER_ZONE > 1, on prend la cible 0 (la plus proche). */
191 const int idx = zone * (int)VL53L8CX_NB_TARGET_PER_ZONE;
192 uint16_t dist = (uint16_t)(dev->results.distance_mm[idx]);
193 if (dist == 0 || dist > 4000) dist = 0;
194 matrix->data[row][col_offset + col] = dist;
195 }
196#else
197 /* ── Mode STUB : scan 8×8 réaliste pour validation visuelle ──────────
198 *
199 * Simule un arc 180° vu depuis un capteur plafond/mur :
200 * · Fond (murs/sol) à 2500–4000 mm selon l'angle et la hauteur
201 * · Distance plus courte aux rangées basses (sol plus proche)
202 * · Variation angulaire cohérente sur 32 colonnes
203 * · Bruit de mesure réaliste ±30 mm
204 *
205 * Valeurs dans la plage exploitable par la heatmap inferno :
206 * 100 mm (très proche / chute) → 4000 mm (max portée)
207 */
208 static int32_t s_stub_seed = 0;
209 s_stub_seed++; /* Compteur pour variation temporelle légère */
210
211 for (int row = 0; row < LIDAR_ROWS; row++) {
212 for (int col = 0; col < 8; col++) {
213 /* Position angulaire dans l'arc 180°
214 * col_offset + col ∈ [0..31] → angle ∈ [0°..180°] */
215 float angle_deg = (float)(col_offset + col) * (180.0f / (LIDAR_COLS - 1));
216 float angle_rad = angle_deg * 3.14159f / 180.0f;
217
218 /* Distance de fond : mur à ~3500 mm en face (90°),
219 * murs latéraux plus proches (0° et 180°) ≈ 2000 mm */
220 float bg_mm = 2000.0f + 1500.0f * sinf(angle_rad);
221
222 /* Facteur hauteur : rangées basses (7) = sol, plus proche */
223 float height_factor = 1.0f - (float)row / (float)(LIDAR_ROWS - 1) * 0.25f;
224 bg_mm *= height_factor;
225
226 /* Bruit de mesure réaliste ±25 mm */
227 int32_t noise = ((s_stub_seed * 1103515245 + 12345) & 0x7FFF) % 50 - 25;
228 s_stub_seed = s_stub_seed * 6364136223846793005LL + 1;
229
230 uint16_t dist = (uint16_t)(bg_mm + (float)noise);
231 if (dist < 100) dist = 100;
232 if (dist > 4000) dist = 4000;
233
234 matrix->data[row][col_offset + col] = dist;
235 }
236 }
237#endif /* LEXA_LIDAR_STUB */
238}

240/* ================================================================
241 * task_sensor_acq (interne)
242 * @brief Tâche FreeRTOS d'acquisition capteurs — Core 1, priorité 10.
243 *
244 * Boucle cadencée à 5 Hz (vTaskDelayUntil) — portée maximale VL53L8CX.
245 * Appelle radar_driver_poll() à chaque itération.
246 * Pousse la sensor_frame_t dans sensor_to_ai_queue (non bloquant).
247 *
248 * @param pvParam Pointeur vers lidar_task_ctx_t alloué par lidar_driver_start.
249 * ================================================================ */
250static void task_sensor_acq(void *pvParam)
251{
252 lidar_task_ctx_t *ctx = (lidar_task_ctx_t *)pvParam;
253
254 /* Abonnement au Task Watchdog Timer (TWDT) */
255 ESP_ERROR_CHECK(esp_task_wdt_add(NULL));
256 ESP_LOGI(TAG, "Task_Sensor_Acq démarrée sur Core %d", xPortGetCoreID());
257
258 /* ================================================================
259 * Initialisation capteurs (déplacée ici pour éviter de bloquer app_main)
260 * ================================================================ */
261 ESP_LOGI(TAG, "Initialisation LIDARs (ULD) dans la tâche capteurs…");
262 for (int i = 0; i < LIDAR_NUM_FRONT; i++) {
263 ctx->devs[i].sensor_idx = i;
264
265 if (!ctx->sys_ctx->lidar_ok[i]) {
266 ctx->devs[i].active = false;
267 ESP_LOGW(TAG, "LIDAR[%d] absent (hw_diag) — ignoré", i);
268 continue;
269 }
270
271 /* Nourrir le watchdog entre capteurs */
272 ESP_ERROR_CHECK(esp_task_wdt_reset());
273 vTaskDelay(pdMS_TO_TICKS(10));
274
275 if (!lidar_init_sensor(&ctx->devs[i], ctx->sys_ctx->lidar_spi[i], i)) {
276 ESP_LOGW(TAG, "LIDAR[%d] : initialisation SPI/ULD échouée", i);
277 ctx->devs[i].active = false;
278 }
279 }
280
281 /* Radar : uniquement si connecté */
282#if LEXACARE_ENABLE_RADAR
283 ESP_ERROR_CHECK(radar_driver_init());
284#else
285 ESP_LOGI(TAG, "Radar LD6002 désactivé (LEXACARE_ENABLE_RADAR=0)");
286#endif
287
288 TickType_t last_wake = xTaskGetTickCount();
289 uint32_t frame_idx = 0;
290
291 while (true) {
292 /* Reset du watchdog avant toute opération bloquante */
293 ESP_ERROR_CHECK(esp_task_wdt_reset());
294
295 sensor_frame_t frame;
296 memset(&frame, 0, sizeof(frame));
297 frame.lidar.timestamp_us = esp_timer_get_time();
298
299 /* Lecture des 4 capteurs frontaux (fusion dans la matrice) */
300 for (int i = 0; i < LIDAR_NUM_FRONT; i++) {
301 lidar_read_sensor(&ctx->devs[i], &frame.lidar, i * 8);
302 }
303
304 frame.lidar.valid = true;
305 frame.lidar_valid = true;
306
307 /* Lecture du radar (non bloquant) — uniquement si connecté */
308#if LEXACARE_ENABLE_RADAR
309 frame.radar_valid = radar_driver_poll(&frame.radar);
310#else
311 frame.radar_valid = false;
312#endif
313
314 /* Envoi dans la queue AI (non bloquant) */
315 if (xQueueSend(ctx->sys_ctx->sensor_to_ai_queue, &frame, 0) != pdTRUE) {
316 ESP_LOGD(TAG, "sensor_to_ai_queue pleine — trame abandonnée");
317 }
318
319 /* ── Mise à jour du snapshot partagé (protégé par mutex) ─────── */
320 if (ctx->sys_ctx->data_mutex &&
321 xSemaphoreTake(ctx->sys_ctx->data_mutex, pdMS_TO_TICKS(5)) == pdTRUE) {
322 ctx->sys_ctx->latest_sensor = frame;
323 ctx->sys_ctx->sensor_valid = true;
324 xSemaphoreGive(ctx->sys_ctx->data_mutex);
325 }
326
327 /* ── Lecture capteurs environnementaux (1 Hz) ─────────────────── */
328 frame_idx++;
329 if (frame_idx >= ENVIRO_READ_PERIOD_FRAMES) {
330 frame_idx = 0;
331 enviro_data_t env = {
332 .temp_hdc_c = NAN, .humidity_hdc = NAN,
333 .temp_bme_c = NAN, .pressure_hpa = NAN,
334 .humidity_bme = NAN,
335 };
336
337 /* HDC1080 */
338 float t_hdc = 0.0f, h_hdc = 0.0f;
339 if (hdc1080_read(&t_hdc, &h_hdc) == ESP_OK) {
340 env.temp_hdc_c = t_hdc;
341 env.humidity_hdc = h_hdc;
342 }
343
344 /* BME280 */
345 float t_bme = 0.0f, p_bme = 0.0f, h_bme = 0.0f;
346 if (bme280_read(&t_bme, &p_bme, &h_bme) == ESP_OK) {
347 env.temp_bme_c = t_bme;
348 env.pressure_hpa = p_bme;
349 env.humidity_bme = h_bme;
350 }
351
352 /* Microphone — burst rapide */
353 uint32_t rms = 0, peak = 0;
354 bool mic_ok = (mic_driver_read(&rms, &peak) == ESP_OK);
355
356 /* Mise à jour sys_context_t */
357 if (ctx->sys_ctx->data_mutex &&
358 xSemaphoreTake(ctx->sys_ctx->data_mutex, pdMS_TO_TICKS(10)) == pdTRUE) {
359 ctx->sys_ctx->enviro = env;
360 ctx->sys_ctx->mic.rms = rms;
361 ctx->sys_ctx->mic.peak = peak;
362 ctx->sys_ctx->mic.valid = mic_ok;
363 xSemaphoreGive(ctx->sys_ctx->data_mutex);
364 }
365 }
366
367 /* Cadence 15 Hz via vTaskDelayUntil (drift-free) */
368 vTaskDelayUntil(&last_wake, pdMS_TO_TICKS(TASK_PERIOD_MS));
369 }
370}

372/* ================================================================
373 * lidar_driver_start
374 * @brief Initialise les capteurs et crée Task_Sensor_Acq sur Core 1.
375 *
376 * @param ctx Pointeur vers le contexte système (bus I2C + lidar_ok[]).
377 * @return ESP_OK si la tâche est créée avec succès.
378 * ================================================================ */
379esp_err_t lidar_driver_start(sys_context_t *ctx)
380{
381 lidar_task_ctx_t *task_ctx = calloc(1, sizeof(lidar_task_ctx_t));
382 if (!task_ctx) {
383 return ESP_ERR_NO_MEM;
384 }
385 task_ctx->sys_ctx = ctx;
386
387 /* Création de la tâche épinglée sur Core 1 */
388 BaseType_t ret = xTaskCreatePinnedToCore(
389 task_sensor_acq,
390 "Task_Sensor_Acq",
391 TASK_STACK_SIZE,
392 task_ctx,
393 TASK_PRIORITY,
394 NULL,
395 TASK_CORE);
396
397 if (ret != pdPASS) {
398 free(task_ctx);
399 ESP_LOGE(TAG, "Échec création Task_Sensor_Acq");
400 return ESP_FAIL;
401 }
402
403 ESP_LOGI(TAG, "Task_Sensor_Acq créée (Core %d, priorité %d, %d Hz)",
404 TASK_CORE, TASK_PRIORITY, 1000 / TASK_PERIOD_MS);
405 return ESP_OK;
406}
