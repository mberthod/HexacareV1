28/**
29 * @brief Swap endianess par mots 32 bits (implémentation typique ST).
30 *
31 * L'ULD manipule des buffers bruts (endianness capteur) et attend que la
32 * plateforme fournisse cette primitive.
33 */
34static inline void VL53L8CX_SwapBuffer(uint8_t *buffer, uint16_t size)
35{
36 if (!buffer) return;
37
38 /* Swap par blocs de 4 octets (uint32_t) */
39 for (uint16_t i = 0; (uint16_t)(i + 3U) < size; i = (uint16_t)(i + 4U)) {
40 const uint8_t b0 = buffer[i + 0U];
41 const uint8_t b1 = buffer[i + 1U];
42 buffer[i + 0U] = buffer[i + 3U];
43 buffer[i + 1U] = buffer[i + 2U];
44 buffer[i + 2U] = b1;
45 buffer[i + 3U] = b0;
46 }
47}
